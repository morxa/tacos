20template <typename LocationT, typename ActionT>
21proto::TimedAutomaton::Transition
22transition_to_proto(const Transition<LocationT, ActionT> &transition)
23{
24 using utilities::to_string;
25 proto::TimedAutomaton::Transition proto;
26 proto.set_source(to_string(transition.source_));
27 proto.set_symbol(transition.symbol_);
28 proto.set_target(to_string(transition.target_));
29 *proto.mutable_clock_resets() = {std::begin(transition.clock_resets_),
30 std::end(transition.clock_resets_)};
31 for (const auto &[clock, constraint] : transition.clock_constraints_) {
32 proto.mutable_clock_constraints()->Add(clock_constraint_to_proto(clock, constraint));
33 }

39template <typename LocationT, typename ActionT>
40proto::TimedAutomaton
41ta_to_proto(const TimedAutomaton<LocationT, ActionT> &ta)
42{
43 using utilities::to_string;
44 proto::TimedAutomaton proto;
45 for (const auto &location : ta.get_locations()) {
46 proto.mutable_locations()->Add(to_string(location));
47 }
48 for (const auto &location : ta.get_final_locations()) {
49 proto.mutable_final_locations()->Add(to_string(location));
50 }
51 proto.set_initial_location(to_string(ta.get_initial_location()));
52 *proto.mutable_alphabet() = {std::begin(ta.get_alphabet()), std::end(ta.get_alphabet())};
53 *proto.mutable_clocks() = {std::begin(ta.get_clocks()), std::end(ta.get_clocks())};
54 for (const auto &[source, transition] : ta.get_transitions()) {
55 proto.mutable_transitions()->Add(details::transition_to_proto(transition));
56 }
57 return proto;
58}
