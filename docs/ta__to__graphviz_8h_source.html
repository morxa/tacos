23template <typename LocationT, typename ActionT>
24utilities::graphviz::Graph
25ta_to_graphviz(const automata::ta::TimedAutomaton<LocationT, ActionT> &ta,
26 bool show_node_labels = true)
27{
28 utilities::graphviz::Graph g;
29 if (!show_node_labels) {
30 g.set_default_node_property("shape", "point");
31 }
32 std::map<automata::ta::Location<LocationT>, utilities::graphviz::Node> nodes;
33 auto initial_node = g.add_node("");
34 initial_node.set_property("shape", "none");
35 for (const auto &location : ta.get_locations()) {
36 std::stringstream str;
37 // Use stream operator of the location.
38 str << location;
39 nodes.insert(std::make_pair(location, g.add_node(str.str())));
40 }
41 g.add_edge(initial_node, nodes[ta.get_initial_location()]);
42 for (const auto &final_location : ta.get_final_locations()) {
43 nodes[final_location].set_property("peripheries", "2");
44 }
45 for (const auto &[source, transition] : ta.get_transitions()) {
46 std::stringstream edge;
47 edge << " " << transition.symbol_ << " \n " << transition.clock_constraints_ << " \n "
48 << fmt::format("{{{}}}", fmt::join(transition.clock_resets_, ", ")) << " ";
49 g.add_edge(nodes[transition.source_], nodes[transition.target_], edge.str());
50 }
51 return g;
52}
