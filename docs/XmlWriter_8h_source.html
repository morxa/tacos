22 inline void
23 add_to_uppaal_xml(const std::pair<std::string, automata::ClockConstraint> &guard,
24  tinyxml2::XMLDocument & doc,
25  tinyxml2::XMLElement * transition_element)
26 {
27  std::stringstream ss;
28  auto * xml_guard = doc.NewElement("label");
29  xml_guard->SetAttribute("kind", "guard");
30  ss << guard.first;
31  if (std::holds_alternative<automata::AtomicClockConstraintT<std::less<Time>>>(guard.second)) {
32  ss << " < ";
33  } else if (std::holds_alternative<automata::AtomicClockConstraintT<std::less_equal<Time>>>(
34  guard.second)) {
35  ss << " <= ";
36  } else if (std::holds_alternative<automata::AtomicClockConstraintT<std::equal_to<Time>>>(
37  guard.second)) {
38  ss << " == ";
39  } else if (std::holds_alternative<automata::AtomicClockConstraintT<std::greater_equal<Time>>>(
40  guard.second)) {
41  ss << " >= ";
42  } else if (std::holds_alternative<automata::AtomicClockConstraintT<std::greater<Time>>>(
43  guard.second)) {
44  ss << " > ";
45  } else {
46  throw automata::InvalidClockComparisonOperatorException();
47  }
48  std::visit(
49  [&ss](const auto atomic_clock_constraint) { ss << atomic_clock_constraint.get_comparand(); },
50  guard.second);
51  xml_guard->SetText(ss.str().c_str());
52  transition_element->InsertEndChild(xml_guard);
53 }

68 template <typename LocationT, typename ActionT>
69 void
70 add_to_uppaal_xml(const automata::ta::Transition<LocationT, ActionT> &transition,
71  tinyxml2::XMLDocument & doc,
72  tinyxml2::XMLElement * ta_element,
73  bool master)
74 {
75  tinyxml2::XMLElement *xml_transition = doc.NewElement("transition");
76  // set source
77  tinyxml2::XMLElement *source = doc.NewElement("source");
78  std::stringstream ss;
79  ss << transition.get_source();
80  source->SetAttribute("ref", ss.str().c_str());
81  xml_transition->InsertEndChild(source);
82  // target
83  // clear stringstream
84  ss.str(std::string());
85  ss << transition.get_target();
86  tinyxml2::XMLElement *target = doc.NewElement("target");
87  target->SetAttribute("ref", ss.str().c_str());
88  xml_transition->InsertEndChild(target);
89  // set guard constraints
90  for (const auto &guard_constraint : transition.get_guards()) {
91  add_to_uppaal_xml(guard_constraint, doc, xml_transition);
92  }
93  // set clock resets
94  {
95  for (const auto &clock_reset : transition.get_reset()) {
96  auto *xml_reset = doc.NewElement("label");
97  xml_reset->SetAttribute("kind", "assignment");
98  // clear stringstream
99  ss.str(std::string());
100  ss << clock_reset << " := 0";
101  xml_reset->SetText(ss.str().c_str());
102  xml_transition->InsertEndChild(xml_reset);
103  }
104  }
105 
106  // set synchronization labels
107  {
108  std::string direction = master ? "!" : "?";
109  for (const auto &label : transition.get_label()) {
110  // clear stringstream
111  ss.str(std::string());
112  auto *xml_label = doc.NewElement("label");
113  xml_label->SetAttribute("kind", "synchronization");
114  ss << label << direction;
115  xml_label->SetText(ss.str().c_str());
116  xml_transition->InsertEndChild(xml_label);
117  }
118  }
119  // add to the ta
120  ta_element->InsertEndChild(xml_transition);
121 }

162 template <typename LocationT, typename ActionT>
163 tinyxml2::XMLElement *
164 add_to_uppaal_xml(const automata::ta::TimedAutomaton<LocationT, ActionT> &ta,
165  tinyxml2::XMLDocument & doc,
166  tinyxml2::XMLElement * root,
167  std::string name,
168  bool master)
169 {
170  tinyxml2::XMLElement *res = doc.NewElement("template");
171  // set the name of the automaton
172  tinyxml2::XMLElement *xml_name = doc.NewElement("name");
173  xml_name->SetText(name.c_str());
174  res->InsertEndChild(xml_name);
175 
176  // declaration: declare clocks
177  tinyxml2::XMLElement *xml_declaration = doc.NewElement("declaration");
178  xml_declaration->SetText(
179  fmt::format("clock {};",
180  fmt::join(std::begin(ta.get_clocks()), std::end(ta.get_clocks()), ", "))
181  .c_str());
182  res->InsertEndChild(xml_declaration);
183 
184  // add locations
185  for (const auto &loc : ta.get_locations()) {
186  add_to_uppaal_xml<LocationT, ActionT>(loc, doc, res);
187  }
188 
189  // set initial location
190  tinyxml2::XMLElement *xml_initial_location = doc.NewElement("init");
191  std::stringstream ss;
192  ss << ta.get_initial_location();
193  xml_initial_location->SetAttribute("ref", ss.str().c_str());
194  res->InsertEndChild(xml_initial_location);
195 
196  // add transitions
197  for (const auto &[source, transition] : ta.get_transitions()) {
198  add_to_uppaal_xml(transition, doc, res, master);
199  }
200 
201  root->InsertEndChild(res);
202  return res;
203 }

220 template <typename LocationT, typename ActionT>
221 void
222 write_composition_to_uppaal(
223  std::string filename,
224  const automata::ta::TimedAutomaton<LocationT, ActionT> & master,
225  const std::vector<automata::ta::TimedAutomaton<LocationT, ActionT>> &slaves)
226 {
227  tinyxml2::XMLDocument doc;
228  // set root node
229  tinyxml2::XMLElement *root = doc.NewElement("nta");
230  doc.InsertEndChild(root);
231  // add master
232  tinyxml2::XMLElement *master_ptr = add_to_uppaal_xml(master, doc, root, "master", true);
233  // add slaves
234  std::vector<tinyxml2::XMLElement *> slave_ptrs;
235  std::size_t count{1};
236  for (const auto &slave : slaves) {
237  slave_ptrs.push_back(
238  add_to_uppaal_xml(slave, doc, root, "slave" + std::to_string(count++), false));
239  }
240  // set up system definition
241  std::vector<std::string> component_names;
242  tinyxml2::XMLElement * sys_ptr = doc.NewElement("system");
243  // compose instantiation-string
244  std::stringstream ss;
245  ss << "M = " << master_ptr->FirstChildElement("name")->GetText() << "();\n";
246  component_names.push_back("M");
247  count = 1;
248  for (auto *slave_ptr : slave_ptrs) {
249  ss << "S" << std::to_string(count) << " = " << slave_ptr->FirstChildElement("name")->GetText()
250  << "();\n";
251  component_names.push_back("S" + std::to_string(count++));
252  }
253 
254  ss << fmt::format("\nsystem {};\n",
255  fmt::join(std::begin(component_names), std::end(component_names), ", "));
256  sys_ptr->SetText(ss.str().c_str());
257  root->InsertEndChild(sys_ptr);
258 
259  // add specification: It should not happen, that all automata are in a final state at the same
260  // time.
261  std::vector<std::set<automata::ta::Location<LocationT>>> final_locations;
262  // collect final locations for all components
263  final_locations.push_back(master.get_final_locations());
264  for (const automata::ta::TimedAutomaton<LocationT, ActionT> &slave : slaves) {
265  final_locations.push_back(slave.get_final_locations());
266  }
267  doc.SaveFile(filename.c_str());
268 }
