16template <class Comp>
17std::ostream &
18operator<<(std::ostream &os, const automata::AtomicClockConstraintT<Comp> &constraint)
19{
20 using automata::InvalidClockComparisonOperatorException;
21
22 if constexpr (std::is_same_v<Comp, std::less<Time>>) {
23 os << "<";
24 } else if constexpr (std::is_same_v<Comp, std::less_equal<Time>>) {
25 os << u8"≤";
26 } else if constexpr (std::is_same_v<Comp, std::equal_to<Time>>) {
27 os << "=";
28 } else if constexpr (std::is_same_v<Comp, std::not_equal_to<Time>>) {
29 os << u8"≠";
30 } else if constexpr (std::is_same_v<Comp, std::greater_equal<Time>>) {
31 os << u8"≥";
32 } else if constexpr (std::is_same_v<Comp, std::greater<Time>>) {
33 os << ">";
34 } else {
35 throw InvalidClockComparisonOperatorException();
36 }
37 os << " " << constraint.comparand_;
38 return os;
39}
