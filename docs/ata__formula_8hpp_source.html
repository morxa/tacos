137 template <typename LocationT>
138 std::set<std::set<State<LocationT>>>
139 ConjunctionFormula<LocationT>::get_minimal_models(const ClockValuation &v) const
140 {
141  auto s1 = conjunct1_->get_minimal_models(v);
142  auto s2 = conjunct2_->get_minimal_models(v);
143  auto cartesian = ranges::views::cartesian_product(s1, s2);
144  std::set<std::set<State<LocationT>>> res;
145  ranges::for_each(cartesian, [&](const auto &prod) {
146  auto u = std::get<0>(prod);
147  u.insert(std::get<1>(prod).begin(), std::get<1>(prod).end());
148  res.insert(u);
149  });
150  return res;
151 }

168 template <typename LocationT>
169 std::set<std::set<State<LocationT>>>
170 DisjunctionFormula<LocationT>::get_minimal_models(const ClockValuation &v) const
171 {
172  auto disjunct1_models = disjunct1_->get_minimal_models(v);
173  auto disjunct2_models = disjunct2_->get_minimal_models(v);
174  // Remove each model from disjunct1_models that is a superset of a model in disjunct2_models
175  std::experimental::erase_if(disjunct1_models, [&disjunct2_models](const auto &model1) {
176  return std::find_if(disjunct2_models.begin(),
177  disjunct2_models.end(),
178  [&model1](const auto &model2) {
179  // True if model1 is a superset of model2.
180  return std::includes(model1.begin(),
181  model1.end(),
182  model2.begin(),
183  model2.end());
184  })
185  != disjunct2_models.end();
186  });
187  // Add each model from disjunct2_models which is not a superset of a model from disjunct1_models.
188  for (const auto &model2 : disjunct2_models) {
189  if (std::find_if(disjunct1_models.begin(),
190  disjunct1_models.end(),
191  [&model2](const auto &model1) {
192  // True if model2 is a superset of model1.
193  return std::includes(model2.begin(),
194  model2.end(),
195  model1.begin(),
196  model1.end());
197  })
198  == disjunct1_models.end()) {
199  disjunct1_models.insert(model2);
200  }
201  }
202  return disjunct1_models;
203 }

234 template <typename LocationT>
235 bool
236 operator<(const Formula<LocationT> &first, const Formula<LocationT> &second)
237 {
238  if (typeid(first) != typeid(second)) {
239  return std::type_index(typeid(first)) < std::type_index(typeid(second));
240  }
241  if (typeid(first) == typeid(TrueFormula<LocationT>)) {
242  return false;
243  }
244  if (typeid(first) == typeid(FalseFormula<LocationT>)) {
245  return false;
246  }
247  if (typeid(first) == typeid(ConjunctionFormula<LocationT>)) {
248  const ConjunctionFormula<LocationT> &f1 =
249  static_cast<const ConjunctionFormula<LocationT> &>(first);
250  const ConjunctionFormula<LocationT> &f2 =
251  static_cast<const ConjunctionFormula<LocationT> &>(second);
252  return std::tie(*f1.conjunct1_, *f1.conjunct2_) < std::tie(*f2.conjunct1_, *f2.conjunct2_);
253  }
254  if (typeid(first) == typeid(DisjunctionFormula<LocationT>)) {
255  const DisjunctionFormula<LocationT> &f1 =
256  static_cast<const DisjunctionFormula<LocationT> &>(first);
257  const DisjunctionFormula<LocationT> &f2 =
258  static_cast<const DisjunctionFormula<LocationT> &>(second);
259  return std::tie(*f1.disjunct1_, *f1.disjunct2_) < std::tie(*f2.disjunct1_, *f2.disjunct2_);
260  }
261  if (typeid(first) == typeid(LocationFormula<LocationT>)) {
262  const LocationFormula<LocationT> &f1 = static_cast<const LocationFormula<LocationT> &>(first);
263  const LocationFormula<LocationT> &f2 = static_cast<const LocationFormula<LocationT> &>(second);
264  return f1.location_ < f2.location_;
265  }
266  if (typeid(first) == typeid(ClockConstraintFormula<LocationT>)) {
267  const ClockConstraintFormula<LocationT> &f1 =
268  static_cast<const ClockConstraintFormula<LocationT> &>(first);
269  const ClockConstraintFormula<LocationT> &f2 =
270  static_cast<const ClockConstraintFormula<LocationT> &>(second);
271  return f1.constraint_ < f2.constraint_;
272  }
273  if (typeid(first) == typeid(ResetClockFormula<LocationT>)) {
274  const ResetClockFormula<LocationT> &f1 =
275  static_cast<const ResetClockFormula<LocationT> &>(first);
276  const ResetClockFormula<LocationT> &f2 =
277  static_cast<const ResetClockFormula<LocationT> &>(second);
278  return *f1.sub_formula_ < *f2.sub_formula_;
279  }
280  throw std::logic_error("Unexpected formulas in comparison");
281 }

312  std::move(conjunct2));
