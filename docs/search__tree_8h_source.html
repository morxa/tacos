82  void
83  set_label(NodeLabel new_label, bool cancel_children = false)
84  {
85  assert(new_label != NodeLabel::UNLABELED);
86  // Check if the node has been labeled before. This is an error, unless either the old or the new
87  // label is CANCELED. This is okay, as we may try to cancel a node that has been labeled in the
88  // meantime (or vice versa).
89  if (label != NodeLabel::UNLABELED && label != NodeLabel::CANCELED
90  && new_label != NodeLabel::CANCELED && label != new_label) {
91  throw std::logic_error(fmt::format(
92  "Trying to set node label to {}, but it is already set to {}", new_label, label));
93  }
94  if (label == NodeLabel::UNLABELED) {
95  SPDLOG_DEBUG(
96  "Labeling {} {} with {}, reason: {}", fmt::ptr(this), *this, new_label, label_reason);
97  label = new_label;
98  if (cancel_children) {
99  for (const auto &action_child : children) {
100  auto child = std::get<1>(action_child);
101  if (std::all_of(std::begin(child->parents),
102  std::end(child->parents),
103  [&child](const auto &parent) {
104  return parent == child.get() || parent->label != NodeLabel::UNLABELED;
105  })) {
106  child->set_label(NodeLabel::CANCELED, true);
107  }
108  }
109  }
110  }
111  }

139  void
140  label_propagate(const std::set<ActionType> &controller_actions,
141  const std::set<ActionType> &environment_actions,
142  bool cancel_children = false)
143  {
144  // SPDLOG_TRACE("Call propagate on node {}", *this);
145  if (is_expanding) {
146  SPDLOG_DEBUG("Cancelling node propagation on {}, currently expanding", *this);
147  return;
148  }
149  if (!is_expanded) {
150  SPDLOG_DEBUG("Cancelling node propagation on {}, node is not expanded yet", *this);
151  return;
152  }
153  // leaf-nodes should always be labelled directly
154  assert(!children.empty() || label != NodeLabel::UNLABELED);
155  // if not already happened: call recursively on parent node
156  if (children.empty()) {
157  assert(label != NodeLabel::UNLABELED);
158  SPDLOG_TRACE("Node is a leaf, propagate labels.", *this);
159  for (const auto &parent : parents) {
160  if (parent != this) {
161  parent->label_propagate(controller_actions, environment_actions, cancel_children);
162  }
163  }
164  return;
165  }
166  // do nothing if the node is already labelled
167  if (label != NodeLabel::UNLABELED) {
168  SPDLOG_TRACE("Node is already labelled, abort.");
169  return;
170  }
171  assert(!children.empty());
172 
173  // Find good and bad child nodes which are already labelled and determine their order (with
174  // respect to time). Also keep track of yet unlabelled nodes (both cases, environmental and
175  // controller action).
176  constexpr auto max = std::numeric_limits<RegionIndex>::max();
177  RegionIndex first_good_controller_step{max};
178  RegionIndex first_non_bad_controller_step{max};
179  RegionIndex first_non_good_environment_step{max};
180  RegionIndex first_bad_environment_step{max};
181  bool has_enviroment_step{false};
182  for (const auto &[timed_action, child] : children) {
183  // Copy label to avoid races while checking the conditions below.
184  const NodeLabel child_label = child->label;
185  const auto &[step, action] = timed_action;
186  if (controller_actions.find(action) != std::end(controller_actions)) {
187  if (child_label == NodeLabel::TOP || child.get() == this) {
188  first_good_controller_step = std::min(first_good_controller_step, step);
189  } else if (child_label == NodeLabel::UNLABELED) {
190  first_non_bad_controller_step = std::min(first_non_bad_controller_step, step);
191  }
192  } else if (environment_actions.find(action) != std::end(environment_actions)) {
193  has_enviroment_step = true;
194  if (child_label == NodeLabel::BOTTOM) {
195  first_bad_environment_step = std::min(first_bad_environment_step, step);
196  } else if (child.get() != this && child_label == NodeLabel::UNLABELED) {
197  first_non_good_environment_step = std::min(first_non_good_environment_step, step);
198  }
199  }
200  }
201  SPDLOG_TRACE("First good ctl step at {}, "
202  "first non-bad ctl step at {}, "
203  "first non-good env step at {}, "
204  "first bad env step at {}",
205  first_good_controller_step,
206  first_non_bad_controller_step,
207  first_non_good_environment_step,
208  first_bad_environment_step);
209 
210  if (first_good_controller_step
211  < std::min(first_bad_environment_step, first_non_good_environment_step)) {
212  // The controller can just select the good controller action.
213  label_reason = LabelReason::GOOD_CONTROLLER_ACTION_FIRST;
214  set_label(NodeLabel::TOP);
215  } else if (has_enviroment_step
216  && std::min(first_bad_environment_step, first_non_good_environment_step)
217  == std::numeric_limits<RegionIndex>::max()) {
218  // There is an environment action and no environment action is bad
219  // -> the controller can just select all environment actions
220  label_reason = LabelReason::NO_BAD_ENV_ACTION;
221  set_label(NodeLabel::TOP);
222  } else if (!has_enviroment_step && first_good_controller_step == max
223  && first_non_bad_controller_step == max) {
224  // All controller actions must be bad (otherwise we would be in the first case)
225  // -> no controller strategy
226  label_reason = LabelReason::ALL_CONTROLLER_ACTIONS_BAD;
227  set_label(NodeLabel::BOTTOM);
228  } else if (has_enviroment_step
229  && first_bad_environment_step
230  < std::min(first_good_controller_step, first_non_bad_controller_step)) {
231  // There must be an environment action (otherwise case 3) and one of them must be bad
232  // (otherwise case 2).
233  assert(first_bad_environment_step < std::numeric_limits<RegionIndex>::max());
234  label_reason = LabelReason::BAD_ENV_ACTION_FIRST;
235  set_label(NodeLabel::BOTTOM);
236  }
237  if (label != NodeLabel::UNLABELED) {
238  for (const auto &parent : parents) {
239  if (parent != this) {
240  parent->label_propagate(controller_actions, environment_actions, cancel_children);
241  }
242  }
243  }
244  }

274  add_child(const std::pair<RegionIndex, ActionType> &action, std::shared_ptr<SearchTreeNode> node)

284  }
