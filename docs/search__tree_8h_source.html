82 void
83 set_label(NodeLabel new_label, bool cancel_children = false)
84 {
85 assert(new_label != NodeLabel::UNLABELED);
86 // Check if the node has been labeled before. This is an error, unless either the old or the new
87 // label is CANCELED. This is okay, as we may try to cancel a node that has been labeled in the
88 // meantime (or vice versa).
89 // TODO load label only once and store locally
90 if (label != NodeLabel::UNLABELED && label != NodeLabel::CANCELED
91 && new_label != NodeLabel::CANCELED && label != new_label) {
92 throw std::logic_error(fmt::format(
93 "Trying to set node label to {}, but it is already set to {}", new_label, label.load()));
94 }
95 if (label == NodeLabel::UNLABELED) {
96 SPDLOG_DEBUG(
97 "Labeling {} {} with {}, reason: {}", fmt::ptr(this), *this, new_label, label_reason);
98 label = new_label;
99 if (cancel_children) {
100 for (const auto &action_child : children) {
101 auto child = std::get<1>(action_child);
102 if (std::all_of(std::begin(child->parents),
103 std::end(child->parents),
104 [&child](const auto &parent) {
105 return parent == child.get() || parent->label != NodeLabel::UNLABELED;
106 })) {
107 child->set_label(NodeLabel::CANCELED, true);
108 }
109 }
110 }
111 }
112 }

140 void
141 label_propagate(const std::set<ActionType> &controller_actions,
142 const std::set<ActionType> &environment_actions,
143 bool cancel_children = false)
144 {
145 // SPDLOG_TRACE("Call propagate on node {}", *this);
146 if (is_expanding) {
147 SPDLOG_DEBUG("Cancelling node propagation on {}, currently expanding", *this);
148 return;
149 }
150 if (!is_expanded) {
151 SPDLOG_DEBUG("Cancelling node propagation on {}, node is not expanded yet", *this);
152 return;
153 }
154 // leaf-nodes should always be labelled directly
155 assert(!children.empty() || label != NodeLabel::UNLABELED);
156 // if not already happened: call recursively on parent node
157 if (children.empty()) {
158 assert(label != NodeLabel::UNLABELED);
159 SPDLOG_TRACE("Node is a leaf, propagate labels.", *this);
160 for (const auto &parent : parents) {
161 if (parent != this) {
162 parent->label_propagate(controller_actions, environment_actions, cancel_children);
163 }
164 }
165 return;
166 }
167 // do nothing if the node is already labelled
168 if (label != NodeLabel::UNLABELED) {
169 SPDLOG_TRACE("Node is already labelled, abort.");
170 return;
171 }
172 assert(!children.empty());
173
174 // Find good and bad child nodes which are already labelled and determine their order (with
175 // respect to time). Also keep track of yet unlabelled nodes (both cases, environmental and
176 // controller action).
177 constexpr auto max = std::numeric_limits<RegionIndex>::max();
178 RegionIndex first_good_controller_step{max};
179 RegionIndex first_non_bad_controller_step{max};
180 RegionIndex first_non_good_environment_step{max};
181 RegionIndex first_bad_environment_step{max};
182 bool has_enviroment_step{false};
183 for (const auto &[timed_action, child] : children) {
184 // Copy label to avoid races while checking the conditions below.
185 const NodeLabel child_label = child->label;
186 const auto &[step, action] = timed_action;
187 if (controller_actions.find(action) != std::end(controller_actions)) {
188 if (child_label == NodeLabel::TOP || child.get() == this) {
189 first_good_controller_step = std::min(first_good_controller_step, step);
190 } else if (child_label == NodeLabel::UNLABELED) {
191 first_non_bad_controller_step = std::min(first_non_bad_controller_step, step);
192 }
193 } else if (environment_actions.find(action) != std::end(environment_actions)) {
194 has_enviroment_step = true;
195 if (child_label == NodeLabel::BOTTOM) {
196 first_bad_environment_step = std::min(first_bad_environment_step, step);
197 } else if (child.get() != this && child_label == NodeLabel::UNLABELED) {
198 first_non_good_environment_step = std::min(first_non_good_environment_step, step);
199 }
200 }
201 }
202 SPDLOG_TRACE("First good ctl step at {}, "
203 "first non-bad ctl step at {}, "
204 "first non-good env step at {}, "
205 "first bad env step at {}",
206 first_good_controller_step,
207 first_non_bad_controller_step,
208 first_non_good_environment_step,
209 first_bad_environment_step);
210
211 if (first_good_controller_step
212 < std::min(first_bad_environment_step, first_non_good_environment_step)) {
213 // The controller can just select the good controller action.
214 label_reason = LabelReason::GOOD_CONTROLLER_ACTION_FIRST;
215 set_label(NodeLabel::TOP, cancel_children);
216 } else if (has_enviroment_step
217 && std::min(first_bad_environment_step, first_non_good_environment_step)
218 == std::numeric_limits<RegionIndex>::max()) {
219 // There is an environment action and no environment action is bad
220 // -> the controller can just select all environment actions
221 label_reason = LabelReason::NO_BAD_ENV_ACTION;
222 set_label(NodeLabel::TOP, cancel_children);
223 } else if (!has_enviroment_step && first_good_controller_step == max
224 && first_non_bad_controller_step == max) {
225 // All controller actions must be bad (otherwise we would be in the first case)
226 // -> no controller strategy
227 label_reason = LabelReason::ALL_CONTROLLER_ACTIONS_BAD;
228 set_label(NodeLabel::BOTTOM, cancel_children);
229 } else if (has_enviroment_step && first_bad_environment_step < max
230 && first_bad_environment_step
231 <= std::min(first_good_controller_step, first_non_bad_controller_step)) {
232 // There must be an environment action (otherwise case 3) and one of them must be bad
233 // (otherwise case 2).
234 assert(first_bad_environment_step < std::numeric_limits<RegionIndex>::max());
235 label_reason = LabelReason::BAD_ENV_ACTION_FIRST;
236 set_label(NodeLabel::BOTTOM, cancel_children);
237 }
238 if (label != NodeLabel::UNLABELED) {
239 for (const auto &parent : parents) {
240 if (parent != this) {
241 parent->label_propagate(controller_actions, environment_actions, cancel_children);
242 }
243 }
244 }
245 }

274 void
275 add_child(const std::pair<RegionIndex, ActionType> &action, std::shared_ptr<SearchTreeNode> node)
276 {
277 if (!children.insert(std::make_pair(action, node)).second) {
278 throw std::invalid_argument(fmt::format("\n{}\nCannot add child node \n{}\n, node already "
279 "has child \n{}\n with the same action ({}, {})",
280 *this,
281 *node,
282 *children.at(action),
283 action.first,
284 action.second));
285 }
286 node->min_total_region_increments =
287 std::min(node->min_total_region_increments, min_total_region_increments + action.first);
288 node->parents.insert(this);
289 }
