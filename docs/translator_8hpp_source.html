80 get_closure(const MTLFormula<ConstraintSymbolT> &formula)
81 {

84  untils.insert(dual_untils.begin(), dual_untils.end());

89 template <typename ConstraintSymbolT>
90 std::unique_ptr<Formula<ConstraintSymbolT>>
91 create_contains(TimeInterval duration)
92 {
93  std::unique_ptr<Formula<ConstraintSymbolT>> lowerBound =
94  std::make_unique<TrueFormula<ConstraintSymbolT>>();
95  std::unique_ptr<Formula<ConstraintSymbolT>> upperBound =
96  std::make_unique<TrueFormula<ConstraintSymbolT>>();
97  if (duration.lowerBoundType() != BoundType::INFTY) {
98  if (duration.lowerBoundType() == BoundType::WEAK) {
99  lowerBound = std::make_unique<ClockConstraintFormula<ConstraintSymbolT>>(
100  AtomicClockConstraintT<std::greater_equal<TimePoint>>(duration.lower()));
101  } else {
102  lowerBound = std::make_unique<ClockConstraintFormula<ConstraintSymbolT>>(
103  AtomicClockConstraintT<std::greater<TimePoint>>(duration.lower()));
104  }
105  }
106  if (duration.upperBoundType() != BoundType::INFTY) {
107  if (duration.upperBoundType() == BoundType::WEAK) {
108  upperBound = std::make_unique<ClockConstraintFormula<ConstraintSymbolT>>(
109  AtomicClockConstraintT<std::less_equal<TimePoint>>(duration.upper()));
110  } else {
111  upperBound = std::make_unique<ClockConstraintFormula<ConstraintSymbolT>>(
112  AtomicClockConstraintT<std::less<TimePoint>>(duration.upper()));
113  }
114  }
115  return ata::create_conjunction(std::move(lowerBound), std::move(upperBound));
116 }

149 template <typename ConstraintSymbolT>
150 std::unique_ptr<Formula<ConstraintSymbolT>>
151 init(const MTLFormula<ConstraintSymbolT> & formula,
152  const AtomicProposition<ConstraintSymbolT> &ap,
153  bool first = false)
154 {
155  switch (formula.get_operator()) {
156  case LOP::TRUE: return std::make_unique<TrueFormula<ConstraintSymbolT>>();
157  case LOP::FALSE: return std::make_unique<FalseFormula<ConstraintSymbolT>>();
158  case LOP::LUNTIL:
159  case LOP::LDUNTIL:
160  // init(psi, a) = x.psi if psi \in cl(phi)
161  if (first) {
162  return std::make_unique<LocationFormula<ConstraintSymbolT>>(formula);
163  } else {
164  return std::make_unique<ResetClockFormula<ConstraintSymbolT>>(
165  std::make_unique<LocationFormula<ConstraintSymbolT>>(formula));
166  }
167  case LOP::LAND:
168  // init(psi_1 AND psi_2, a) = init(psi_1, a) AND init(psi_2, a)
169  return ata::create_conjunction(init(formula.get_operands().front(), ap, first),
170  init(formula.get_operands().back(), ap, first));
171  case LOP::LOR:
172  // init(psi_1 OR psi_2, a) = init(psi_1, a) OR init(psi_2, a)
173  return ata::create_disjunction(init(formula.get_operands().front(), ap, first),
174  init(formula.get_operands().back(), ap, first));
175  case LOP::AP:
176  if (formula == ap) {
177  // init(b, a) = TRUE if b == a
178  return std::make_unique<TrueFormula<ConstraintSymbolT>>();
179  } else {
180  // init(b, a) = FALSE if b != a
181  return std::make_unique<FalseFormula<ConstraintSymbolT>>();
182  }
183  case LOP::LNEG:
184  // init(NOT b, a) = NOT init(b, a)
185  // ATA formulas do not have negations, directly compute the result.
186  // We know that this is an atomic proposition because the input formula is in positive normal
187  // form.
188  switch (formula.get_operands().front().get_operator()) {
189  case LOP::TRUE: return std::make_unique<FalseFormula<ConstraintSymbolT>>();
190  case LOP::FALSE: return std::make_unique<TrueFormula<ConstraintSymbolT>>();
191  case LOP::AP:
192  if (formula.get_operands().front() == ap) {
193  // init(not b, a) = FALSE if b == a
194  return std::make_unique<FalseFormula<ConstraintSymbolT>>();
195  } else {
196  // init(not b, a) = TRUE if b != a
197  return std::make_unique<TrueFormula<ConstraintSymbolT>>();
198  }
199  default:
200  std::stringstream ss;
201  ss << "The formula " << formula << " is not in positive normal form.";
202  throw std::logic_error(ss.str());
203  }
204  }
205  throw std::logic_error("Unexpected formula operator");
206 }

252 template <typename ConstraintSymbolT>
253 AlternatingTimedAutomaton<ConstraintSymbolT>
254 translate(const MTLFormula<ConstraintSymbolT> & input_formula,
255  std::set<AtomicProposition<ConstraintSymbolT>> alphabet)
256 {
257  const auto formula = input_formula.to_positive_normal_form();
258  if (alphabet.empty()) {
259  // The ATA alphabet is the same as the formula alphabet.
260  alphabet = formula.get_alphabet();
261  }
262  if (alphabet.count(get_l0<ConstraintSymbolT>()) > 0) {
263  throw std::invalid_argument(fmt::format("The formula alphabet must not contain the symbol '{}'",
264  get_l0<ConstraintSymbolT>()));
265  }
266  if (alphabet.count(get_sink<ConstraintSymbolT>()) > 0) {
267  throw std::invalid_argument(fmt::format("The formula alphabet must not contain the symbol '{}'",
268  get_sink<ConstraintSymbolT>()));
269  }
270  // S = cl(phi) U {l0}
271  auto locations = get_closure(formula);
272  locations.insert(get_l0<ConstraintSymbolT>());
273  const auto untils = formula.get_subformulas_of_type(LOP::LUNTIL);
274  const auto dual_untils = formula.get_subformulas_of_type(LOP::LDUNTIL);
275  const auto accepting_locations = dual_untils;
276  std::set<Transition<ConstraintSymbolT>> transitions;
277  for (const auto &symbol : alphabet) {
278  // Initial transition delta(l0, symbol) -> phi
279  transitions.insert(Transition<ConstraintSymbolT>(get_l0<ConstraintSymbolT>(),
280  symbol.ap_,
281  init(formula, symbol, true)));
282 
283  for (const auto &until : untils) {
284  auto transition_formula = ata::create_disjunction(
285  // init(phi_2, symbol) AND x \in I
286  ata::create_conjunction(init(until.get_operands().back(), symbol),
287  create_contains<ConstraintSymbolT>(until.get_interval())),
288  // init(phi_1, symbol) AND (psi_1 \until_I psi_2)
289  // \-> == until <-/
290  ata::create_conjunction(init(until.get_operands().front(), symbol),
291  std::unique_ptr<Formula<ConstraintSymbolT>>(
292  std::make_unique<LocationFormula<ConstraintSymbolT>>(until))));
293  transitions.insert(
294  Transition<ConstraintSymbolT>(until, symbol, std::move(transition_formula)));
295  }
296  for (const auto &dual_until : dual_untils) {
297  auto transition_formula = ata::create_conjunction(
298  ata::create_disjunction(init(dual_until.get_operands().back(), symbol),
299  create_negated_contains<ConstraintSymbolT>(
300  dual_until.get_interval())),
301  ata::create_disjunction(init(dual_until.get_operands().front(), symbol),
302  std::unique_ptr<Formula<ConstraintSymbolT>>(
303  std::make_unique<LocationFormula<ConstraintSymbolT>>(
304  dual_until))));
305  transitions.insert(
306  Transition<ConstraintSymbolT>(dual_until, symbol, std::move(transition_formula)));
307  }
308  }
309  return AlternatingTimedAutomaton<ConstraintSymbolT>(
310  alphabet,
311  MTLFormula<ConstraintSymbolT>{get_l0<ConstraintSymbolT>()},
312  accepting_locations,
313  std::move(transitions),
314  MTLFormula<ConstraintSymbolT>{get_sink<ConstraintSymbolT>()});
315 }
