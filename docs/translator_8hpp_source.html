75 template <typename ConstraintSymbolT>
76 std::unique_ptr<Formula<ConstraintSymbolT>>
77 create_contains(TimeInterval duration)
78 {
79  std::unique_ptr<Formula<ConstraintSymbolT>> lowerBound =
80  std::make_unique<TrueFormula<ConstraintSymbolT>>();
81  std::unique_ptr<Formula<ConstraintSymbolT>> upperBound =
82  std::make_unique<TrueFormula<ConstraintSymbolT>>();
83  if (duration.lowerBoundType() != BoundType::INFTY) {
84  if (duration.lowerBoundType() == BoundType::WEAK) {
85  lowerBound = std::make_unique<ClockConstraintFormula<ConstraintSymbolT>>(
86  AtomicClockConstraintT<std::greater_equal<Time>>(duration.lower()));
87  } else {
88  lowerBound = std::make_unique<ClockConstraintFormula<ConstraintSymbolT>>(
89  AtomicClockConstraintT<std::greater<Time>>(duration.lower()));
90  }
91  }
92  if (duration.upperBoundType() != BoundType::INFTY) {
93  if (duration.upperBoundType() == BoundType::WEAK) {
94  upperBound = std::make_unique<ClockConstraintFormula<ConstraintSymbolT>>(
95  AtomicClockConstraintT<std::less_equal<Time>>(duration.upper()));
96  } else {
97  upperBound = std::make_unique<ClockConstraintFormula<ConstraintSymbolT>>(
98  AtomicClockConstraintT<std::less<Time>>(duration.upper()));
99  }
100  }
101  return ata::create_conjunction(std::move(lowerBound), std::move(upperBound));
102 }

135 template <typename ConstraintSymbolT, typename SymbolT, bool state_based>
136 std::unique_ptr<Formula<ConstraintSymbolT>>
137 init(const MTLFormula<ConstraintSymbolT> &formula,
138  const AtomicProposition<SymbolT> &ap,
139  bool first = false)
140 {
141  static_assert(state_based || std::is_same_v<SymbolT, ConstraintSymbolT>);
142  static_assert(!state_based || std::is_same_v<SymbolT, std::set<ConstraintSymbolT>>);
143  switch (formula.get_operator()) {
144  case LOP::TRUE: return std::make_unique<TrueFormula<ConstraintSymbolT>>();
145  case LOP::FALSE: return std::make_unique<FalseFormula<ConstraintSymbolT>>();
146  case LOP::LUNTIL:
147  case LOP::LDUNTIL:
148  // init(psi, a) = x.psi if psi \in cl(phi)
149  if (first) {
150  return std::make_unique<LocationFormula<ConstraintSymbolT>>(formula);
151  } else {
152  return std::make_unique<ResetClockFormula<ConstraintSymbolT>>(
153  std::make_unique<LocationFormula<ConstraintSymbolT>>(formula));
154  }
155  case LOP::LAND: {
156  std::vector<std::unique_ptr<Formula<ConstraintSymbolT>>> conjuncts;
157  for (const auto &conjunct : formula.get_operands()) {
158  conjuncts.push_back(init<ConstraintSymbolT, SymbolT, state_based>(conjunct, ap, first));
159  }
160  return ata::create_conjunction(std::move(conjuncts));
161  }
162  case LOP::LOR: {
163  // init(psi_1 OR psi_2, a) = init(psi_1, a) OR init(psi_2, a)
164  std::vector<std::unique_ptr<Formula<ConstraintSymbolT>>> disjuncts;
165  for (const auto &disjunct : formula.get_operands()) {
166  disjuncts.push_back(init<ConstraintSymbolT, SymbolT, state_based>(disjunct, ap, first));
167  }
168  return ata::create_disjunction(std::move(disjuncts));
169  }
170  case LOP::AP:
171  if constexpr (state_based) {
172  if (ap.ap_.find(formula.get_atomicProposition().ap_) != ap.ap_.end()) {
173  // init(b, a) = TRUE if b is contained in a
174  return std::make_unique<TrueFormula<ConstraintSymbolT>>();
175  } else {
176  // init(b, a) = FALSE if b is not contained in a
177  return std::make_unique<FalseFormula<ConstraintSymbolT>>();
178  }
179  } else {
180  if (formula == ap) {
181  // init(b, a) = TRUE if b == a
182  return std::make_unique<TrueFormula<ConstraintSymbolT>>();
183  } else {
184  // init(b, a) = FALSE if b != a
185  return std::make_unique<FalseFormula<ConstraintSymbolT>>();
186  }
187  }
188  case LOP::LNEG:
189  // init(NOT b, a) = NOT init(b, a)
190  // ATA formulas do not have negations, directly compute the result.
191  // We know that this is an atomic proposition because the input formula is in positive normal
192  // form.
193  switch (formula.get_operands().front().get_operator()) {
194  case LOP::TRUE: return std::make_unique<FalseFormula<ConstraintSymbolT>>();
195  case LOP::FALSE: return std::make_unique<TrueFormula<ConstraintSymbolT>>();
196  case LOP::AP:
197  if constexpr (state_based) {
198  if (ap.ap_.find(formula.get_operands().front().get_atomicProposition().ap_)
199  != ap.ap_.end()) {
200  // init(b, a) = FALSE if b is contained in a
201  return std::make_unique<FalseFormula<ConstraintSymbolT>>();
202  } else {
203  // init(b, a) = TRUE if b is not contained in a
204  return std::make_unique<TrueFormula<ConstraintSymbolT>>();
205  }
206  } else {
207  if (formula.get_operands().front() == ap) {
208  // init(b, a) = TRUE if b == a
209  return std::make_unique<FalseFormula<ConstraintSymbolT>>();
210  } else {
211  // init(b, a) = FALSE if b != a
212  return std::make_unique<TrueFormula<ConstraintSymbolT>>();
213  }
214  }
215  default:
216  std::stringstream ss;
217  ss << "The formula " << formula << " is not in positive normal form.";
218  throw std::logic_error(ss.str());
219  }
220  }
221  throw std::logic_error("Unexpected formula operator");
222 }

245 template <bool state_based, typename ConstraintSymbolT>
246 auto
247 compute_alphabet(std::set<AtomicProposition<ConstraintSymbolT>> input_alphabet)
248 {
249  auto unwrap = [](std::set<AtomicProposition<ConstraintSymbolT>> input) {
250  std::set<ConstraintSymbolT> res;
251  for (const auto &i : input) {
252  res.insert(i.ap_);
253  }
254  return res;
255  };
256  auto wrap = [](std::set<std::set<ConstraintSymbolT>> input) {
257  std::set<AtomicProposition<std::set<ConstraintSymbolT>>> res;
258  for (const auto &i : input) {
259  res.insert(AtomicProposition{i});
260  }
261  return res;
262  };
263 
264  if constexpr (state_based) {
265  return wrap(utilities::construct_powerset(unwrap(input_alphabet)));
266  } else {
267  return input_alphabet;
268  }

277 template <typename ConstraintSymbolT, typename SymbolT, bool state_based>
278 ATA<ConstraintSymbolT, SymbolT>
279 translate(const MTLFormula<ConstraintSymbolT> &input_formula,
280  std::set<AtomicProposition<SymbolT>> input_alphabet)
281 {
282  const auto formula = input_formula.to_positive_normal_form();
283  if constexpr (state_based) {
284  if (input_alphabet.empty()) {
285  input_alphabet = compute_alphabet<true>(formula.get_alphabet());
286  }
287  if (input_alphabet.count(AtomicProposition<SymbolT>{{get_l0<ConstraintSymbolT>().ap_}}) > 0) {
288  throw std::invalid_argument(
289  fmt::format("The formula alphabet must not contain the symbol '{{{}}}'",
290  get_l0<ConstraintSymbolT>()));
291  }
292  if (input_alphabet.count(AtomicProposition<SymbolT>{{get_sink<ConstraintSymbolT>().ap_}}) > 0) {
293  throw std::invalid_argument(
294  fmt::format("The formula alphabet must not contain the symbol '{{{}}}'",
295  get_sink<ConstraintSymbolT>()));
296  }
297  } else {
298  if (input_alphabet.empty()) {
299  // The ATA alphabet is the same as the formula alphabet.
300  input_alphabet = formula.get_alphabet();
301  }
302  if (input_alphabet.count(get_l0<ConstraintSymbolT>()) > 0) {
303  throw std::invalid_argument(
304  fmt::format("The formula alphabet must not contain the symbol '{}'",
305  get_l0<ConstraintSymbolT>()));
306  }
307  if (input_alphabet.count(get_sink<ConstraintSymbolT>()) > 0) {
308  throw std::invalid_argument(
309  fmt::format("The formula alphabet must not contain the symbol '{}'",
310  get_sink<ConstraintSymbolT>()));
311  }
312  }
313  const auto alphabet = input_alphabet;
314  // const auto alphabet = compute_alphabet<state_based, ConstraintSymbolT>(input_alphabet);
315  // S = cl(phi) U {l0}
316  auto locations = get_closure(formula);
317  locations.insert(get_l0<ConstraintSymbolT>());
318  const auto untils = formula.get_subformulas_of_type(LOP::LUNTIL);
319  const auto dual_untils = formula.get_subformulas_of_type(LOP::LDUNTIL);
320  const auto accepting_locations = dual_untils;
321  std::set<T<ConstraintSymbolT, SymbolT>> transitions;
322  for (const auto &symbol : alphabet) {
323  // Initial transition delta(l0, symbol) -> phi
324  transitions.insert(T<ConstraintSymbolT, SymbolT>(
325  get_l0<ConstraintSymbolT>(),
326  symbol,
327  init<ConstraintSymbolT, SymbolT, state_based>(formula, symbol, true)));
328 
329  for (const auto &until : untils) {
330  auto transition_formula = ata::create_disjunction(
331  // init(phi_2, symbol) AND x \in I
332  ata::create_conjunction(
333  init<ConstraintSymbolT, SymbolT, state_based>(until.get_operands().back(), symbol),
334  create_contains<ConstraintSymbolT>(until.get_interval())),
335  // init(phi_1, symbol) AND (psi_1 \until_I psi_2)
336  // \-> == until <-/
337  ata::create_conjunction(
338  init<ConstraintSymbolT, SymbolT, state_based>(until.get_operands().front(), symbol),
339  std::unique_ptr<Formula<ConstraintSymbolT>>(
340  std::make_unique<LocationFormula<ConstraintSymbolT>>(until))));
341  transitions.insert(
342  T<ConstraintSymbolT, SymbolT>(until, symbol, std::move(transition_formula)));
343  }
344  for (const auto &dual_until : dual_untils) {
345  auto transition_formula = ata::create_conjunction(
346  ata::create_disjunction(
347  init<ConstraintSymbolT, SymbolT, state_based>(dual_until.get_operands().back(), symbol),
348  create_negated_contains<ConstraintSymbolT>(dual_until.get_interval())),
349  ata::create_disjunction(
350  init<ConstraintSymbolT, SymbolT, state_based>(dual_until.get_operands().front(), symbol),
351  std::unique_ptr<Formula<ConstraintSymbolT>>(
352  std::make_unique<LocationFormula<ConstraintSymbolT>>(dual_until))));
353  transitions.insert(
354  T<ConstraintSymbolT, SymbolT>(dual_until, symbol, std::move(transition_formula)));
355  }
356  }
357  return ATA<ConstraintSymbolT, SymbolT>(alphabet,
358  MTLFormula<ConstraintSymbolT>{get_l0<ConstraintSymbolT>()},
359  accepting_locations,
360  std::move(transitions),
361  MTLFormula<ConstraintSymbolT>{
362  get_sink<ConstraintSymbolT>()});
363 }
