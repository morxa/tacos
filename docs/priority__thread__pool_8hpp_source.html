48 template <class Priority, class T>
49 void
50 ThreadPool<Priority, T>::start()
51 {
52  if (started) {
53  throw QueueStartedException("Pool already started");
54  }
55  worker_idle = std::vector(size, false);
56  for (std::size_t i = 0; i < size; ++i) {
57  workers.push_back(std::thread{[this, i]() {
58  while (!stopping) {
59  {
60  std::lock_guard idle_guard{worker_idle_mutex};
61  worker_idle[i] = false;
62  }
63  std::unique_lock lock{queue_mutex};
64  while (!queue.empty()) {
65  auto job = std::get<1>(queue.top());
66  queue.pop();
67  lock.unlock();
68  job();
69  lock.lock();
70  if (stopping) {
71  return;
72  }
73  }
74  {
75  std::lock_guard done_guard{worker_idle_mutex};
76  worker_idle[i] = true;
77  worker_idle_cond.notify_all();
78  }
79  if (!queue_open) {
80  return;
81  }
82  // Wait for the stop signal or a new job.
83  queue_cond.wait(lock, [this] { return stopping || !queue.empty() || !queue_open; });
84  }
85  }});
86  }
87  started = true;
88 }

123 void
124 ThreadPool<Priority, T>::wait()
125 {
126  std::unique_lock lock{worker_idle_mutex};
127  while (true) {
128  if (std::all_of(begin(worker_idle), end(worker_idle), [](const auto &worker_idle) {
129  return worker_idle;
130  })) {
131  return;
132  }
133  worker_idle_cond.wait(lock);
134  }
