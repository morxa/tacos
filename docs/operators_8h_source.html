32 bool
33 is_monotonically_dominated(const CanonicalABWord<LocationT, ConstraintSymbolT> &w1,
34  const CanonicalABWord<LocationT, ConstraintSymbolT> &w2)
35 {
36  auto current_w2_it = w2.begin();
37  for (const auto &w1_partition : w1) {
38  // Find the w2 partition that includes the current w1 partition.
39  const auto next_w2_it =
40  std::find_if(current_w2_it, w2.end(), [&w1_partition](const auto &w2_partition) {
41  return std::includes(w2_partition.begin(),
42  w2_partition.end(),
43  w1_partition.begin(),
44  w1_partition.end());
45  });
46  if (next_w2_it == w2.end()) {
47  return false;
48  }
49  current_w2_it = std::next(next_w2_it);
50  }
51  return true;
52 }

90 ancestor_is_monotonically_dominated(

92  const std::set<CanonicalABWord<LocationT, ConstraintSymbolT>> & words,

94 {

96  return false;

98  seen_nodes.push_back(&node);
99  return is_monotonically_dominated(node.words, words)

103  return ancestor_is_monotonically_dominated(*parent, words, seen_nodes);
