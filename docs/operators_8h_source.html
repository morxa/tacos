31template <typename LocationT, typename ConstraintSymbolT>
32bool
33is_monotonically_dominated(const CanonicalABWord<LocationT, ConstraintSymbolT> &w1,
34 const CanonicalABWord<LocationT, ConstraintSymbolT> &w2)
35{
36 auto current_w2_it = w2.begin();
37 for (const auto &w1_partition : w1) {
38 // Find the w2 partition that includes the current w1 partition.
39 const auto next_w2_it =
40 std::find_if(current_w2_it, w2.end(), [&w1_partition](const auto &w2_partition) {
41 return std::includes(w2_partition.begin(),
42 w2_partition.end(),
43 w1_partition.begin(),
44 w1_partition.end());
45 });
46 if (next_w2_it == w2.end()) {
47 return false;
48 }
49 current_w2_it = std::next(next_w2_it);
50 }
51 return true;
52}

63template <typename LocationT, typename ConstraintSymbolT>
64bool
65is_monotonically_dominated(const std::set<CanonicalABWord<LocationT, ConstraintSymbolT>> &set1,
66 const std::set<CanonicalABWord<LocationT, ConstraintSymbolT>> &set2)
67{
68 return std::all_of(set2.begin(), set2.end(), [&set1](const auto &word2) {
69 return std::any_of(set1.begin(), set1.end(), [&word2](const auto &word1) {
70 return is_monotonically_dominated(word1, word2);
71 });
72 });
73}

88template <typename LocationT, typename ActionT, typename ConstraintSymbolT>
89bool
90ancestor_is_monotonically_dominated(
91 const SearchTreeNode<LocationT, ActionT, ConstraintSymbolT> & node,
92 const std::set<CanonicalABWord<LocationT, ConstraintSymbolT>> & words,
93 std::vector<const SearchTreeNode<LocationT, ActionT, ConstraintSymbolT> *> &seen_nodes)
94{
95 if (std::find(std::begin(seen_nodes), std::end(seen_nodes), &node) != std::end(seen_nodes)) {
96 return false;
97 }
98 seen_nodes.push_back(&node);
99 return is_monotonically_dominated(node.words, words)
100 || std::any_of(node.parents.begin(),
101 node.parents.end(),
102 [&words, &seen_nodes](const auto &parent) {
103 return ancestor_is_monotonically_dominated(*parent, words, seen_nodes);
104 });
105 return false;
106}

112bool
113dominates_ancestor(SearchTreeNode<LocationT, ActionT, ConstraintSymbolT> *node)
114{
115 std::vector<const SearchTreeNode<LocationT, ActionT, ConstraintSymbolT> *> seen_nodes = {node};
116 return std::any_of(node->parents.begin(),
117 node->parents.end(),
118 [node, &seen_nodes](const auto &parent) {
119 return ancestor_is_monotonically_dominated(*parent, node->words, seen_nodes);
120 });
