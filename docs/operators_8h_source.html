41 template <typename LocationT, typename ConstraintSymbolT>
42 bool
43 is_monotonically_dominated(const CanonicalABWord<LocationT, ConstraintSymbolT> &w1,
44  const CanonicalABWord<LocationT, ConstraintSymbolT> &w2)
45 {
46  auto current_w2_it = w2.begin();
47  for (const auto &w1_partition : w1) {
48  // Find the w2 partition that includes the current w1 partition.
49  const auto next_w2_it =
50  std::find_if(current_w2_it, w2.end(), [&w1_partition](const auto &w2_partition) {
51  return std::includes(w2_partition.begin(),
52  w2_partition.end(),
53  w1_partition.begin(),
54  w1_partition.end());
55  });
56  if (next_w2_it == w2.end()) {
57  return false;
58  }
59  current_w2_it = std::next(next_w2_it);
60  }
61  return true;
62 }

76  const std::set<CanonicalABWord<LocationT, ConstraintSymbolT>> &set2)
77 {
78  return std::all_of(set2.begin(), set2.end(), [&set1](const auto &word2) {
79  return std::any_of(set1.begin(), set1.end(), [&word2](const auto &word1) {
80  return is_monotonically_dominated(word1, word2);

98 template <typename LocationT, typename ActionT, typename ConstraintSymbolT>
99 bool
100 ancestor_is_monotonically_dominated(
101  const SearchTreeNode<LocationT, ActionT, ConstraintSymbolT> & node,
102  const std::set<CanonicalABWord<LocationT, ConstraintSymbolT>> & words,
103  std::vector<const SearchTreeNode<LocationT, ActionT, ConstraintSymbolT> *> &seen_nodes)
104 {
105  if (std::find(std::begin(seen_nodes), std::end(seen_nodes), &node) != std::end(seen_nodes)) {
106  return false;
107  }
108  seen_nodes.push_back(&node);
109  return is_monotonically_dominated(node.words, words)
110  || std::any_of(node.parents.begin(),
111  node.parents.end(),
112  [&words, &seen_nodes](const auto &parent) {
113  return ancestor_is_monotonically_dominated(*parent, words, seen_nodes);
114  });
115  return false;
116 }
