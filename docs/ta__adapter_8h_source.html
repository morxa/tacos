74 ActionType,

78 for (const auto &symbol : ta.get_alphabet()) {
79 SPDLOG_TRACE("({}, {}): Symbol {}", ab_configuration.first, ab_configuration.second, symbol);
80 const std::set<typename automata::ta::TimedAutomaton<LocationT, ActionType>::Configuration>
81 ta_successors = ta.make_symbol_step(ab_configuration.first, symbol);
82 std::set<ATAConfiguration<ConstraintSymbolType>> ata_successors;
83 if constexpr (!use_location_constraints) {
84 ata_successors = ata.make_symbol_step(ab_configuration.second, symbol);
85 }
86 SPDLOG_TRACE("({}, {}): TA successors: {} ATA successors: {}",
87 ab_configuration.first,
88 ab_configuration.second,
89 ta_successors.size(),
90 ata_successors.size());
91 for (const auto &ta_successor : ta_successors) {
92 SPDLOG_TRACE("({}, {}): TA successor {}",
93 ab_configuration.first,
94 ab_configuration.second,
95 ta_successor);
96 if constexpr (use_location_constraints) {
97 ata_successors = ata.make_symbol_step(ab_configuration.second,
98 logic::AtomicProposition{ta_successor.location});
99 }
100 for (const auto &ata_successor : ata_successors) {
101 SPDLOG_TRACE("({}, {}): ATA successor {}",
102 ab_configuration.first,
103 ab_configuration.second,
104 ata_successor);
105 [[maybe_unused]] auto successor = successors.insert(
106 std::make_pair(symbol, get_canonical_word(ta_successor, ata_successor, K)));
107 SPDLOG_TRACE("({}, {}): Getting {} with symbol {}",
108 ab_configuration.first,
109 ab_configuration.second,
110 successor->second,
111 symbol);
112 }
113 }
114 }

116 }
