188inline std::optional<std::size_t>
189get_relation_index(const ClockConstraint &constraint)
190{
191 if (std::holds_alternative<AtomicClockConstraintT<std::less<Time>>>(constraint)) {
192 return 0;
193 } else if (std::holds_alternative<AtomicClockConstraintT<std::less_equal<Time>>>(constraint)) {
194 return 1;
195 } else if (std::holds_alternative<AtomicClockConstraintT<std::equal_to<Time>>>(constraint)) {
196 return 2;
197 } else if (std::holds_alternative<AtomicClockConstraintT<std::not_equal_to<Time>>>(constraint)) {
198 return 3;
199 } else if (std::holds_alternative<AtomicClockConstraintT<std::greater_equal<Time>>>(constraint)) {
200 return 4;
201 } else if (std::holds_alternative<AtomicClockConstraintT<std::greater<Time>>>(constraint)) {
202 return 5;
203 } else {
204 assert(false);
205 return std::nullopt;
206 }
207}

213inline bool
214operator<(const ClockConstraint &lhs, const ClockConstraint &rhs)
215{
216 auto lhs_idx = get_relation_index(lhs);
217 auto rhs_idx = get_relation_index(rhs);
218 if (!lhs_idx || !rhs_idx) {
219 throw std::logic_error("Unknown relation symbol in constraints cannot be handled.");
220 }
221 if (lhs_idx.value() < rhs_idx.value()) {
222 return true;
223 } else if (lhs_idx.value() == rhs_idx.value()) {
224 Time lhs_time = std::visit([](const auto &atomic_clock_constraint)
225 -> Time { return atomic_clock_constraint.get_comparand(); },
226 lhs);
227 Time rhs_time = std::visit([](const auto &atomic_clock_constraint)
228 -> Time { return atomic_clock_constraint.get_comparand(); },
229 rhs);
230
231 return lhs_time < rhs_time;
232 } else {
233 return false;
234 }
235}
