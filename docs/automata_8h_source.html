186 inline std::optional<std::size_t>
187 get_relation_index(const ClockConstraint &constraint)
188 {
189  if (std::holds_alternative<AtomicClockConstraintT<std::less<Time>>>(constraint)) {
190  return 0;
191  } else if (std::holds_alternative<AtomicClockConstraintT<std::less_equal<Time>>>(constraint)) {
192  return 1;
193  } else if (std::holds_alternative<AtomicClockConstraintT<std::equal_to<Time>>>(constraint)) {
194  return 2;
195  } else if (std::holds_alternative<AtomicClockConstraintT<std::not_equal_to<Time>>>(constraint)) {
196  return 3;
197  } else if (std::holds_alternative<AtomicClockConstraintT<std::greater_equal<Time>>>(constraint)) {
198  return 4;
199  } else if (std::holds_alternative<AtomicClockConstraintT<std::greater<Time>>>(constraint)) {
200  return 5;
201  } else {
202  assert(false);
203  return std::nullopt;
204  }
205 }

211 inline bool
212 operator<(const ClockConstraint &lhs, const ClockConstraint &rhs)
213 {
214  auto lhs_idx = get_relation_index(lhs);
215  auto rhs_idx = get_relation_index(rhs);
216  if (!lhs_idx || !rhs_idx) {
217  throw std::logic_error("Unknown relation symbol in constraints cannot be handled.");
218  }
219  if (lhs_idx.value() < rhs_idx.value()) {
220  return true;
221  } else if (lhs_idx.value() == rhs_idx.value()) {
222  Time lhs_time = std::visit([](const auto &atomic_clock_constraint)
223  -> Time { return atomic_clock_constraint.get_comparand(); },
224  lhs);
225  Time rhs_time = std::visit([](const auto &atomic_clock_constraint)
226  -> Time { return atomic_clock_constraint.get_comparand(); },
227  rhs);
228 
229  return lhs_time < rhs_time;
230  } else {
231  return false;
232  }
233 }
