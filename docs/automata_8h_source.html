186 inline std::optional<std::size_t>
187 get_relation_index(const ClockConstraint &constraint)
188 {
189  if (std::holds_alternative<AtomicClockConstraintT<std::less<Time>>>(constraint)) {
190  return 0;
191  } else if (std::holds_alternative<AtomicClockConstraintT<std::less_equal<Time>>>(constraint)) {
192  return 1;
193  } else if (std::holds_alternative<AtomicClockConstraintT<std::equal_to<Time>>>(constraint)) {
194  return 2;
195  } else if (std::holds_alternative<AtomicClockConstraintT<std::not_equal_to<Time>>>(constraint)) {
196  return 3;
197  } else if (std::holds_alternative<AtomicClockConstraintT<std::greater_equal<Time>>>(constraint)) {
198  return 4;
199  } else if (std::holds_alternative<AtomicClockConstraintT<std::greater<Time>>>(constraint)) {
200  return 5;
201  } else {
202  assert(false);
203  return std::nullopt;
204  }
205 }

207 inline bool
208 operator<(const ClockConstraint &lhs, const ClockConstraint &rhs)
209 {
210  auto lhs_idx = get_relation_index(lhs);
211  auto rhs_idx = get_relation_index(rhs);
212  if (!lhs_idx || !rhs_idx) {
213  throw std::logic_error("Unknown relation symbol in constraints cannot be handled.");
214  }
215  if (lhs_idx.value() < rhs_idx.value()) {
216  return true;
217  } else if (lhs_idx.value() == rhs_idx.value()) {
218  Time lhs_time = std::visit([](const auto &atomic_clock_constraint)
219  -> Time { return atomic_clock_constraint.get_comparand(); },
220  lhs);
221  Time rhs_time = std::visit([](const auto &atomic_clock_constraint)
222  -> Time { return atomic_clock_constraint.get_comparand(); },
223  rhs);
224 
225  return lhs_time < rhs_time;
226  } else {
227  return false;
228  }
229 }
