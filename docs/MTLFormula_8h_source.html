36enum class LOP { LAND, LOR, LNEG, LUNTIL, LDUNTIL, AP, TRUE, FALSE };
37
39inline LOP
40dual(LOP in)
41{
42 switch (in) {
43 case LOP::LAND: return LOP::LOR;
44 case LOP::LOR: return LOP::LAND;
45 case LOP::LUNTIL: return LOP::LDUNTIL;
46 case LOP::LDUNTIL: return LOP::LUNTIL;
47 case LOP::TRUE: return LOP::FALSE;
48 case LOP::FALSE: return LOP::TRUE;
49 default: return in;
50 }
51}
