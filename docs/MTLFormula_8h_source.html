48 inline LOP
49 dual(LOP in)
50 {
51  switch (in) {
52  case LOP::LAND: return LOP::LOR;
53  case LOP::LOR: return LOP::LAND;
54  case LOP::LUNTIL: return LOP::LDUNTIL;
55  case LOP::LDUNTIL: return LOP::LUNTIL;
56  case LOP::TRUE: return LOP::FALSE;
57  case LOP::FALSE: return LOP::TRUE;
58  default: return in;
59  }
60 }
