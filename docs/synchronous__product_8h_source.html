38 template <typename Location, typename ConstraintSymbolType>
39 std::set<ABRegionSymbol<Location, ConstraintSymbolType>>
40 increment_region_indexes(
41  const std::set<ABRegionSymbol<Location, ConstraintSymbolType>> &configurations,
42  RegionIndex max_region_index)
43 {
44  // Assert that our assumption holds: All region indexes are either odd or even, never mixed.
45  assert(
46  std::all_of(std::begin(configurations),
47  std::end(configurations),
48  [](const auto &configuration) { return get_region_index(configuration) % 2 == 0; })
49  || std::all_of(std::begin(configurations),
50  std::end(configurations),
51  [](const auto &configuration) {
52  return get_region_index(configuration) % 2 == 1;
53  }));
54  std::set<ABRegionSymbol<Location, ConstraintSymbolType>> res;
55  std::transform(configurations.begin(),
56  configurations.end(),
57  std::inserter(res, res.end()),
58  [max_region_index](auto configuration) {
59  if (std::holds_alternative<PlantRegionState<Location>>(configuration)) {
60  auto &ta_configuration = std::get<PlantRegionState<Location>>(configuration);
61  RegionIndex &region_index = ta_configuration.region_index;
62  // Increment if the region index is less than the max_region_index and if the
63  // region index is even or if we increment all region indexes.
64  if (region_index < max_region_index) {
65  region_index += 1;
66  }
67  } else {
68  auto &ata_configuration =
69  std::get<ATARegionState<ConstraintSymbolType>>(configuration);
70  RegionIndex &region_index = ata_configuration.region_index;
71  // Increment if the region index is less than the max_region_index and if the
72  // region index is even or if we increment all region indexes.
73  if (region_index < max_region_index) {
74  region_index += 1;
75  }
76  }
77  return configuration;
78  });
79  return res;
80 }

93 template <typename Location, typename ConstraintSymbolType>
94 CanonicalABWord<Location, ConstraintSymbolType>
95 get_time_successor(const CanonicalABWord<Location, ConstraintSymbolType> &word, RegionIndex K)
96 {
97  assert(is_valid_canonical_word(word));
98  if (word.empty()) {
99  return {};
100  }
101  CanonicalABWord<Location, ConstraintSymbolType> res;
102  const RegionIndex max_region_index = 2 * K + 1;
103  // Find the partition that contains all maxed partitions. If it does not exist, create an empty
104  // one.
105  std::set<ABRegionSymbol<Location, ConstraintSymbolType>> new_maxed_partition;
106  auto last_nonmaxed_partition = std::next(word.rbegin());
107  // Check if maxed partition is actually maxed
108  if (std::all_of(word.rbegin()->begin(),
109  word.rbegin()->end(),
110  [&max_region_index](const auto &configuration) {
111  return get_region_index(configuration) == max_region_index;
112  })) {
113  new_maxed_partition = *word.rbegin();
114  } else {
115  // There is no maxed partition, so the last partition is already nonmaxed.
116  last_nonmaxed_partition = word.rbegin();
117  }
118  if (last_nonmaxed_partition == word.rend()) {
119  // All partitions are maxed, nothing to increment.
120  return word;
121  }
122  {
123  // Increment the last nonmaxed partition. If we have a new maxed configuration, put it into the
124  // maxed partition. Otherwise, keep it in place.
125  auto incremented = increment_region_indexes(*last_nonmaxed_partition, max_region_index);
126  std::set<ABRegionSymbol<Location, ConstraintSymbolType>> incremented_nonmaxed;
127  for (auto &configuration : incremented) {
128  if (get_region_index(configuration) == max_region_index) {
129  new_maxed_partition.insert(configuration);
130  } else {
131  incremented_nonmaxed.insert(configuration);
132  }
133  }
134  if (!incremented_nonmaxed.empty()) {
135  res.push_back(std::move(incremented_nonmaxed));
136  }
137  }
138 
139  // Process all partitions before the last nonmaxed partition.
140  if (std::prev(std::rend(word)) != last_nonmaxed_partition) {
141  // The first set needs to be incremented if its region indexes are even.
142  if (get_region_index(*word.begin()->begin()) % 2 == 0) {
143  auto incremented = increment_region_indexes(*word.begin(), max_region_index);
144  std::set<ABRegionSymbol<Location, ConstraintSymbolType>> incremented_nonmaxed;
145  for (auto &configuration : incremented) {
146  if (get_region_index(configuration) == max_region_index) {
147  new_maxed_partition.insert(configuration);
148  } else {
149  incremented_nonmaxed.insert(configuration);
150  }
151  }
152  if (!incremented_nonmaxed.empty()) {
153  res.push_back(std::move(incremented_nonmaxed));
154  }
155  } else {
156  res.push_back(*word.begin());
157  }
158  // Copy all other abs_i which are not the first nor the last. Those never change.
159  std::reverse_copy(std::next(last_nonmaxed_partition),
160  std::prev(word.rend()),
161  std::back_inserter(res));
162  }
163  // If the maxed partition is non-empty, add it to the resulting word.
164  if (!new_maxed_partition.empty()) {
165  res.push_back(std::move(new_maxed_partition));
166  }
167  assert(is_valid_canonical_word(res));
168  return res;
169 }

180 template <typename Location /* TA::Location */, typename ConstraintSymbolType>
181 std::pair<PlantConfiguration<Location>, ATAConfiguration<ConstraintSymbolType>>
182 get_candidate(const CanonicalABWord<Location, ConstraintSymbolType> &word)
183 {
184  assert(is_valid_canonical_word(word));
185  PlantConfiguration<Location> plant_configuration{};
186  ATAConfiguration<ConstraintSymbolType> ata_configuration{};
187  const Time time_delta = Time(1) / Time(word.size() + 1);
188  for (std::size_t i = 0; i < word.size(); i++) {
189  const auto &abs_i = word[i];
190  for (const ABRegionSymbol<Location, ConstraintSymbolType> &symbol : abs_i) {
191  // TODO Refactor, fractional/integral outside of if
192  if (std::holds_alternative<PlantRegionState<Location>>(symbol)) {
193  const auto & ta_region_state = std::get<PlantRegionState<Location>>(symbol);
194  const RegionIndex region_index = ta_region_state.region_index;
195  const Time fractional_part = region_index % 2 == 0 ? 0 : time_delta * (i + 1);
196  const Time integral_part = static_cast<RegionIndex>(region_index / 2);
197  const auto & clock_name = ta_region_state.clock;
198  // update ta_configuration
199  plant_configuration.location = ta_region_state.location;
200  plant_configuration.clock_valuations[clock_name] = integral_part + fractional_part;
201  } else { // ATARegionState<ConstraintSymbolType>
202  const auto & ata_region_state = std::get<ATARegionState<ConstraintSymbolType>>(symbol);
203  const RegionIndex region_index = ata_region_state.region_index;
204  const Time fractional_part = region_index % 2 == 0 ? 0 : time_delta * (i + 1);
205  const Time integral_part = static_cast<RegionIndex>(region_index / 2);
206  // update configuration
207  // TODO check: the formula (aka ConstraintSymbolType) encodes the location, the clock
208  // valuation is separate and a configuration is a set of such pairs. Is this already
209  // sufficient?
210  ata_configuration.insert(ATAState<ConstraintSymbolType>{ata_region_state.formula,
211  fractional_part + integral_part});
212  }
213  }
214  }
215  return std::make_pair(plant_configuration, ata_configuration);
216 }

237 template <typename Location, typename ConstraintSymbolType>
238 std::vector<std::pair<RegionIndex, CanonicalABWord<Location, ConstraintSymbolType>>>
239 get_time_successors(const CanonicalABWord<Location, ConstraintSymbolType> &canonical_word,
240  RegionIndex K)
241 {
242  SPDLOG_TRACE("Computing time successors of {} with K={}", canonical_word, K);
243  auto cur = get_time_successor(canonical_word, K);
244  RegionIndex cur_index{0};
245  std::vector<std::pair<RegionIndex, CanonicalABWord<Location, ConstraintSymbolType>>>
246  time_successors;
247  time_successors.push_back(std::make_pair(cur_index++, canonical_word));
248  for (; cur != time_successors.back().second; cur_index++) {
249  time_successors.emplace_back(cur_index, cur);
250  cur = get_time_successor(time_successors.back().second, K);
251  }
252  return time_successors;
253 }
