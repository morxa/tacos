117 template <typename Location, typename ConstraintSymbolType>
118 bool
119 is_valid_canonical_word(const CanonicalABWord<Location, ConstraintSymbolType> &word)
120 {
121  // TODO all ta_symbols should agree on the same location
122  // TODO clocks must have unique values (i.e., must not occur multiple times)
123  if (word.empty()) {
124  throw InvalidCanonicalWordException(word, "word is empty");
125  }
126  // No configuration should be empty
127  if (std::any_of(word.begin(), word.end(), [](const auto &configurations) {
128  return configurations.empty();
129  })) {
130  throw InvalidCanonicalWordException(word, "word contains an empty configuration");
131  }
132  // Each partition either contains only even or only odd region indexes. This is because the word
133  // is partitioned by the fractional part and the region index can only be even if the fractional
134  // part is 0. If that is the case, there cannot be any configuration with an odd region index in
135  // the same partition, as that configuration's fractional part would be > 0.
136  std::for_each(word.begin(), word.end(), [&word](const auto &configurations) {
137  if (std::any_of(configurations.begin(),
138  configurations.end(),
139  [](const auto &w) { return get_region_index(w) % 2 == 0; })
140  && std::any_of(configurations.begin(), configurations.end(), [](const auto &w) {
141  return get_region_index(w) % 2 == 1;
142  })) {
143  throw InvalidCanonicalWordException(word, "both odd and even region indexes");
144  }
145  });
146  // There must be at most one partition with fractional part 0.
147  // The only partition that is allowed to have fracitonal part 0 is the 0th
148  // partition.
149  std::for_each(std::next(word.begin()), word.end(), [&word](const auto &configurations) {
150  std::for_each(configurations.begin(), configurations.end(), [&word](const auto &w) {
151  if (get_region_index(w) % 2 == 0) {
152  throw InvalidCanonicalWordException(word,
153  "fractional part 0 in wrong element of partition");
154  }
155  });
156  });
157  return true;
158 }

166 template <typename Location, typename ConstraintSymbolType>
167 std::set<ABRegionSymbol<Location, ConstraintSymbolType>>
168 increment_region_indexes(
169  const std::set<ABRegionSymbol<Location, ConstraintSymbolType>> &configurations,
170  RegionIndex max_region_index)
171 {
172  // Assert that our assumption holds: All region indexes are either odd or even, never mixed.
173  assert(
174  std::all_of(std::begin(configurations),
175  std::end(configurations),
176  [](const auto &configuration) { return get_region_index(configuration) % 2 == 0; })
177  || std::all_of(std::begin(configurations),
178  std::end(configurations),
179  [](const auto &configuration) {
180  return get_region_index(configuration) % 2 == 1;
181  }));
182  std::set<ABRegionSymbol<Location, ConstraintSymbolType>> res;
183  std::transform(configurations.begin(),
184  configurations.end(),
185  std::inserter(res, res.end()),
186  [max_region_index](auto configuration) {
187  if (std::holds_alternative<TARegionState<Location>>(configuration)) {
188  auto &ta_configuration = std::get<TARegionState<Location>>(configuration);
189  RegionIndex &region_index = ta_configuration.region_index;
190  // Increment if the region index is less than the max_region_index and if the
191  // region index is even or if we increment all region indexes.
192  if (region_index < max_region_index) {
193  region_index += 1;
194  }
195  } else {
196  auto &ata_configuration =
197  std::get<ATARegionState<ConstraintSymbolType>>(configuration);
198  RegionIndex &region_index = ata_configuration.region_index;
199  // Increment if the region index is less than the max_region_index and if the
200  // region index is even or if we increment all region indexes.
201  if (region_index < max_region_index) {
202  region_index += 1;
203  }
204  }
205  return configuration;
206  });
207  return res;
208 }

221 template <typename Location, typename ConstraintSymbolType>
222 CanonicalABWord<Location, ConstraintSymbolType>
223 get_time_successor(const CanonicalABWord<Location, ConstraintSymbolType> &word,
224  automata::ta::Integer K)
225 {
226  assert(is_valid_canonical_word(word));
227  if (word.empty()) {
228  return {};
229  }
230  CanonicalABWord<Location, ConstraintSymbolType> res;
231  const RegionIndex max_region_index = 2 * K + 1;
232  // Find the partition that contains all maxed partitions. If it does not exist, create an empty
233  // one.
234  std::set<ABRegionSymbol<Location, ConstraintSymbolType>> new_maxed_partition;
235  auto last_nonmaxed_partition = std::next(word.rbegin());
236  // Check if maxed partition is actually maxed
237  if (std::all_of(word.rbegin()->begin(),
238  word.rbegin()->end(),
239  [&max_region_index](const auto &configuration) {
240  return get_region_index(configuration) == max_region_index;
241  })) {
242  new_maxed_partition = *word.rbegin();
243  } else {
244  // There is no maxed partition, so the last partition is already nonmaxed.
245  last_nonmaxed_partition = word.rbegin();
246  }
247  if (last_nonmaxed_partition == word.rend()) {
248  // All partitions are maxed, nothing to increment.
249  return word;
250  }
251  {
252  // Increment the last nonmaxed partition. If we have a new maxed configuration, put it into the
253  // maxed partition. Otherwise, keep it in place.
254  auto incremented = increment_region_indexes(*last_nonmaxed_partition, max_region_index);
255  std::set<ABRegionSymbol<Location, ConstraintSymbolType>> incremented_nonmaxed;
256  for (auto &configuration : incremented) {
257  if (get_region_index(configuration) == max_region_index) {
258  new_maxed_partition.insert(configuration);
259  } else {
260  incremented_nonmaxed.insert(configuration);
261  }
262  }
263  if (!incremented_nonmaxed.empty()) {
264  res.push_back(std::move(incremented_nonmaxed));
265  }
266  }
267 
268  // Process all partitions before the last nonmaxed partition.
269  if (std::prev(std::rend(word)) != last_nonmaxed_partition) {
270  // The first set needs to be incremented if its region indexes are even.
271  if (get_region_index(*word.begin()->begin()) % 2 == 0) {
272  auto incremented = increment_region_indexes(*word.begin(), max_region_index);
273  std::set<ABRegionSymbol<Location, ConstraintSymbolType>> incremented_nonmaxed;
274  for (auto &configuration : incremented) {
275  if (get_region_index(configuration) == max_region_index) {
276  new_maxed_partition.insert(configuration);
277  } else {
278  incremented_nonmaxed.insert(configuration);
279  }
280  }
281  if (!incremented_nonmaxed.empty()) {
282  res.push_back(std::move(incremented_nonmaxed));
283  }
284  } else {
285  res.push_back(*word.begin());
286  }
287  // Copy all other abs_i which are not the first nor the last. Those never change.
288  std::reverse_copy(std::next(last_nonmaxed_partition),
289  std::prev(word.rend()),
290  std::back_inserter(res));
291  }
292  // If the maxed partition is non-empty, add it to the resulting word.
293  if (!new_maxed_partition.empty()) {
294  res.push_back(std::move(new_maxed_partition));
295  }
296  assert(is_valid_canonical_word(res));
297  return res;
298 }

315 template <typename Location, typename ConstraintSymbolType>
316 CanonicalABWord<Location, ConstraintSymbolType>
317 get_canonical_word(const automata::ta::Configuration<Location> & ta_configuration,
318  const ATAConfiguration<ConstraintSymbolType> &ata_configuration,
319  const unsigned int K)
320 {
321  // TODO Also accept a TA that does not have any clocks.
322  if (ta_configuration.clock_valuations.empty()) {
323  throw std::invalid_argument("TA without clocks are not supported");
324  }
325  std::set<ABSymbol<Location, ConstraintSymbolType>> g;
326  // Insert ATA configurations into g.
327  std::copy(ata_configuration.begin(), ata_configuration.end(), std::inserter(g, g.end()));
328  // Insert TA configurations into g.
329  for (const auto &[clock_name, clock_value] : ta_configuration.clock_valuations) {
330  g.insert(TAState<Location>{ta_configuration.location, clock_name, clock_value});
331  }
332  // Sort into partitions by the fractional parts.
333  std::map<ClockValuation,
334  std::set<ABSymbol<Location, ConstraintSymbolType>>,
335  utilities::ApproxFloatComparator<Time>>
336  partitioned_g(utilities::ApproxFloatComparator<Time>{});
337  for (const ABSymbol<Location, ConstraintSymbolType> &symbol : g) {
338  partitioned_g[utilities::getFractionalPart<int, ClockValuation>(get_time(symbol))].insert(
339  symbol);
340  }
341  // Replace exact clock values by region indices.
342  automata::ta::TimedAutomatonRegions regionSet{K};
343  CanonicalABWord<Location, ConstraintSymbolType> abs;
344  for (const auto &[fractional_part, g_i] : partitioned_g) {
345  std::set<ABRegionSymbol<Location, ConstraintSymbolType>> abs_i;
346  std::transform(g_i.begin(),
347  g_i.end(),
348  std::inserter(abs_i, abs_i.end()),
349  [&](const ABSymbol<Location, ConstraintSymbolType> &w)
350  -> ABRegionSymbol<Location, ConstraintSymbolType> {
351  if (std::holds_alternative<TAState<Location>>(w)) {
352  const TAState<Location> &s = std::get<TAState<Location>>(w);
353  return TARegionState<Location>{s.location,
354  s.clock,
355  regionSet.getRegionIndex(s.clock_valuation)};
356  } else {
357  const ATAState<ConstraintSymbolType> &s =
358  std::get<ATAState<ConstraintSymbolType>>(w);
359  return ATARegionState<ConstraintSymbolType>{s.location,
360  regionSet.getRegionIndex(
361  s.clock_valuation)};
362  }
363  });
364  abs.push_back(abs_i);
365  }
366  assert(is_valid_canonical_word(abs));
367  return abs;
368 }

379 template <typename Location, typename ConstraintSymbolType>
380 std::pair<TAConfiguration<Location>, ATAConfiguration<ConstraintSymbolType>>
381 get_candidate(const CanonicalABWord<Location, ConstraintSymbolType> &word)
382 {
383  assert(is_valid_canonical_word(word));
384  TAConfiguration<Location> ta_configuration{};
385  ATAConfiguration<ConstraintSymbolType> ata_configuration{};
386  const Time time_delta = Time(1) / Time(word.size() + 1);
387  for (std::size_t i = 0; i < word.size(); i++) {
388  const auto &abs_i = word[i];
389  for (const ABRegionSymbol<Location, ConstraintSymbolType> &symbol : abs_i) {
390  // TODO Refactor, fractional/integral outside of if
391  if (std::holds_alternative<TARegionState<Location>>(symbol)) {
392  const auto & ta_region_state = std::get<TARegionState<Location>>(symbol);
393  const RegionIndex region_index = ta_region_state.region_index;
394  const Time fractional_part = region_index % 2 == 0 ? 0 : time_delta * (i + 1);
395  const Time integral_part = static_cast<RegionIndex>(region_index / 2);
396  const auto & clock_name = ta_region_state.clock;
397  // update ta_configuration
398  ta_configuration.location = ta_region_state.location;
399  ta_configuration.clock_valuations[clock_name] = integral_part + fractional_part;
400  } else { // ATARegionState<ConstraintSymbolType>
401  const auto & ata_region_state = std::get<ATARegionState<ConstraintSymbolType>>(symbol);
402  const RegionIndex region_index = ata_region_state.region_index;
403  const Time fractional_part = region_index % 2 == 0 ? 0 : time_delta * (i + 1);
404  const Time integral_part = static_cast<RegionIndex>(region_index / 2);
405  // update configuration
406  // TODO check: the formula (aka ConstraintSymbolType) encodes the location, the clock
407  // valuation is separate and a configuration is a set of such pairs. Is this already
408  // sufficient?
409  ata_configuration.insert(ATAState<ConstraintSymbolType>{ata_region_state.formula,
410  fractional_part + integral_part});
411  }
412  }
413  }
414  return std::make_pair(ta_configuration, ata_configuration);
415 }

436 template <typename Location, typename ConstraintSymbolType>
437 std::vector<std::pair<RegionIndex, CanonicalABWord<Location, ConstraintSymbolType>>>
438 get_time_successors(const CanonicalABWord<Location, ConstraintSymbolType> &canonical_word,
439  RegionIndex K)
440 {
441  SPDLOG_TRACE("Computing time successors of {} with K={}", canonical_word, K);
442  auto cur = get_time_successor(canonical_word, K);
443  RegionIndex cur_index{0};
444  std::vector<std::pair<RegionIndex, CanonicalABWord<Location, ConstraintSymbolType>>>
445  time_successors;
446  time_successors.push_back(std::make_pair(cur_index++, canonical_word));
447  for (; cur != time_successors.back().second; cur_index++) {
448  time_successors.emplace_back(cur_index, cur);
449  cur = get_time_successor(time_successors.back().second, K);
450  }
451  return time_successors;
452 }

463 get_next_canonical_words(

489  SPDLOG_TRACE("({}, {}): TA successor {}",
490  ab_configuration.first,
491  ab_configuration.second,
492  ta_successor);
493  if constexpr (use_location_constraints) {
494  ata_successors = ata.make_symbol_step(ab_configuration.second,
495  logic::AtomicProposition{ta_successor.location.get()});
496  }
497  for (const auto &ata_successor : ata_successors) {
498  SPDLOG_TRACE("({}, {}): ATA successor {}",
499  ab_configuration.first,
500  ab_configuration.second,
501  ata_successor);
502  res.push_back(get_canonical_word(ta_successor, ata_successor, K));
503  SPDLOG_TRACE("({}, {}): Getting {} with symbol {}",
504  ab_configuration.first,
505  ab_configuration.second,
506  res.back(),
507  symbol);
508  }
509  }
