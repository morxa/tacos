37template <typename Location, typename ConstraintSymbolType>
38std::set<ABRegionSymbol<Location, ConstraintSymbolType>>
39increment_region_indexes(
40 const std::set<ABRegionSymbol<Location, ConstraintSymbolType>> &configurations,
41 RegionIndex max_region_index)
42{
43 // Assert that our assumption holds: All region indexes are either odd or even, never mixed.
44 assert(
45 std::all_of(std::begin(configurations),
46 std::end(configurations),
47 [](const auto &configuration) { return get_region_index(configuration) % 2 == 0; })
48 || std::all_of(std::begin(configurations),
49 std::end(configurations),
50 [](const auto &configuration) {
51 return get_region_index(configuration) % 2 == 1;
52 }));
53 std::set<ABRegionSymbol<Location, ConstraintSymbolType>> res;
54 std::transform(configurations.begin(),
55 configurations.end(),
56 std::inserter(res, res.end()),
57 [max_region_index](auto configuration) {
58 if (std::holds_alternative<PlantRegionState<Location>>(configuration)) {
59 auto &ta_configuration = std::get<PlantRegionState<Location>>(configuration);
60 RegionIndex &region_index = ta_configuration.region_index;
61 // Increment if the region index is less than the max_region_index and if the
62 // region index is even or if we increment all region indexes.
63 if (region_index < max_region_index) {
64 region_index += 1;
65 }
66 } else {
67 auto &ata_configuration =
68 std::get<ATARegionState<ConstraintSymbolType>>(configuration);
69 RegionIndex &region_index = ata_configuration.region_index;
70 // Increment if the region index is less than the max_region_index and if the
71 // region index is even or if we increment all region indexes.
72 if (region_index < max_region_index) {
73 region_index += 1;
74 }
75 }
76 return configuration;
77 });
78 return res;
79}

92template <typename Location, typename ConstraintSymbolType>
93CanonicalABWord<Location, ConstraintSymbolType>
94get_time_successor(const CanonicalABWord<Location, ConstraintSymbolType> &word, RegionIndex K)
95{
96 if (word.empty()) {
97 return {};
98 }
99 CanonicalABWord<Location, ConstraintSymbolType> res;
100 const RegionIndex max_region_index = 2 * K + 1;
101 assert(is_valid_canonical_word(word, max_region_index));
102 // Find the partition that contains all maxed partitions. If it does not exist, create an empty
103 // one.
104 std::set<ABRegionSymbol<Location, ConstraintSymbolType>> new_maxed_partition;
105 auto last_nonmaxed_partition = std::next(word.rbegin());
106 // Check if maxed partition is actually maxed
107 if (std::all_of(word.rbegin()->begin(),
108 word.rbegin()->end(),
109 [&max_region_index](const auto &configuration) {
110 return get_region_index(configuration) == max_region_index;
111 })) {
112 new_maxed_partition = *word.rbegin();
113 } else {
114 // There is no maxed partition, so the last partition is already nonmaxed.
115 last_nonmaxed_partition = word.rbegin();
116 }
117 if (last_nonmaxed_partition == word.rend()) {
118 // All partitions are maxed, nothing to increment.
119 return word;
120 }
121 const bool has_even_region_index = get_region_index(*word.begin()->begin()) % 2 == 0;
122 // The first set needs to be incremented if its region indexes are even.
123 if (has_even_region_index) {
124 auto incremented = increment_region_indexes(*word.begin(), max_region_index);
125 std::set<ABRegionSymbol<Location, ConstraintSymbolType>> incremented_nonmaxed;
126 for (auto &configuration : incremented) {
127 if (get_region_index(configuration) == max_region_index) {
128 new_maxed_partition.insert(configuration);
129 } else {
130 incremented_nonmaxed.insert(configuration);
131 }
132 }
133 if (!incremented_nonmaxed.empty()) {
134 res.push_back(std::move(incremented_nonmaxed));
135 }
136 std::reverse_copy(last_nonmaxed_partition, std::prev(word.rend()), std::back_inserter(res));
137 } else {
138 // Increment the last nonmaxed partition. If we have a new maxed configuration, put it into the
139 // maxed partition. Otherwise, keep it in place.
140 auto incremented = increment_region_indexes(*last_nonmaxed_partition, max_region_index);
141 std::set<ABRegionSymbol<Location, ConstraintSymbolType>> incremented_nonmaxed;
142 for (auto &configuration : incremented) {
143 if (get_region_index(configuration) == max_region_index) {
144 new_maxed_partition.insert(configuration);
145 } else {
146 incremented_nonmaxed.insert(configuration);
147 }
148 }
149 if (!incremented_nonmaxed.empty()) {
150 res.push_back(std::move(incremented_nonmaxed));
151 }
152 std::reverse_copy(std::next(last_nonmaxed_partition), word.rend(), std::back_inserter(res));
153 }
154
155 // If the maxed partition is non-empty, add it to the resulting word.
156 if (!new_maxed_partition.empty()) {
157 res.push_back(std::move(new_maxed_partition));
158 }
159 assert(is_valid_canonical_word(res, max_region_index));
160 return res;
161}

172template <typename Location /* TA::Location */, typename ConstraintSymbolType>
173std::pair<PlantConfiguration<Location>, ATAConfiguration<ConstraintSymbolType>>
174get_candidate(const CanonicalABWord<Location, ConstraintSymbolType> &word)
175{
176 assert(is_valid_canonical_word(word));
177 PlantConfiguration<Location> plant_configuration{};
178 ATAConfiguration<ConstraintSymbolType> ata_configuration{};
179 const Time time_delta = Time(1) / Time(word.size() + 1);
180 for (std::size_t i = 0; i < word.size(); i++) {
181 const auto &abs_i = word[i];
182 for (const ABRegionSymbol<Location, ConstraintSymbolType> &symbol : abs_i) {
183 // TODO Refactor, fractional/integral outside of if
184 if (std::holds_alternative<PlantRegionState<Location>>(symbol)) {
185 const auto &ta_region_state = std::get<PlantRegionState<Location>>(symbol);
186 const RegionIndex region_index = ta_region_state.region_index;
187 const Time fractional_part =
188 region_index % 2 == 0 ? 0 : time_delta * static_cast<Time>((i + 1));
189 const Time integral_part = static_cast<RegionIndex>(region_index / 2);
190 const auto &clock_name = ta_region_state.clock;
191 // update ta_configuration
192 plant_configuration.location = ta_region_state.location;
193 plant_configuration.clock_valuations[clock_name] = integral_part + fractional_part;
194 } else { // ATARegionState<ConstraintSymbolType>
195 const auto &ata_region_state = std::get<ATARegionState<ConstraintSymbolType>>(symbol);
196 const RegionIndex region_index = ata_region_state.region_index;
197 const Time fractional_part =
198 region_index % 2 == 0 ? 0 : time_delta * static_cast<Time>((i + 1));
199 const Time integral_part = static_cast<RegionIndex>(region_index / 2);
200 // update configuration
201 // TODO check: the formula (aka ConstraintSymbolType) encodes the location, the clock
202 // valuation is separate and a configuration is a set of such pairs. Is this already
203 // sufficient?
204 ata_configuration.insert(ATAState<ConstraintSymbolType>{ata_region_state.formula,
205 fractional_part + integral_part});
206 }
207 }
208 }
209 return std::make_pair(plant_configuration, ata_configuration);
210}

231template <typename Location, typename ConstraintSymbolType>
232std::vector<std::pair<RegionIndex, CanonicalABWord<Location, ConstraintSymbolType>>>
233get_time_successors(const CanonicalABWord<Location, ConstraintSymbolType> &canonical_word,
234 RegionIndex K)
235{
236 SPDLOG_TRACE("Computing time successors of {} with K={}", canonical_word, K);
237 auto cur = get_time_successor(canonical_word, K);
238 RegionIndex cur_index{0};
239 std::vector<std::pair<RegionIndex, CanonicalABWord<Location, ConstraintSymbolType>>>
240 time_successors;
241 time_successors.push_back(std::make_pair(cur_index++, canonical_word));
242 for (; cur != time_successors.back().second; cur_index++) {
243 time_successors.emplace_back(cur_index, cur);
244 cur = get_time_successor(time_successors.back().second, K);
245 }
246 return time_successors;
247}
253template <typename Location, typename ConstraintSymbolType>
254std::set<CanonicalABWord<Location, ConstraintSymbolType>>
255get_next_time_successors(
256 const std::set<CanonicalABWord<Location, ConstraintSymbolType>> &canonical_words,
257 RegionIndex K)
258{
259 assert(!canonical_words.empty());
260 assert(std::all_of(std::begin(canonical_words), std::end(canonical_words), [&](const auto &word) {
261 return reg_a(word) == reg_a(*std::begin(canonical_words));
262 }));
263 std::set<CanonicalABWord<Location, ConstraintSymbolType>> successors;
264 std::map<CanonicalABWord<Location, ConstraintSymbolType>,
265 CanonicalABWord<Location, ConstraintSymbolType>>
266 successors_map;
267 for (const auto &word : canonical_words) {
268 successors_map[word] = get_time_successor(word, K);
269 }
270 if (std::any_of(std::begin(successors_map), std::end(successors_map), [](const auto &map_entry) {
271 return reg_a(map_entry.first) == reg_a(map_entry.second);
272 })) {
273 // There is at least one where the successor has the same reg_a and thus there is an ATA
274 // configuration that is incremented. We must only increments those where there is also an ATA
275 // configuration to increment.
276 for (const auto &[word, successor] : successors_map) {
277 if (reg_a(word) == reg_a(successor)) {
278 successors.insert(successor);
279 } else {
280 successors.insert(word);
281 }
282 }
283 } else {
284 for (const auto &[_, successor] : successors_map) {
285 successors.insert(successor);
286 }
287 }
288 return successors;
289}

297template <typename Location, typename ConstraintSymbolType>
298std::vector<std::set<CanonicalABWord<Location, ConstraintSymbolType>>>
299get_time_successors(
300 const std::set<CanonicalABWord<Location, ConstraintSymbolType>> &canonical_words,
301 RegionIndex K)
302{
303 std::vector<std::set<CanonicalABWord<Location, ConstraintSymbolType>>> successors;
304 successors.push_back(canonical_words);
305 while (true) {
306 const auto next = get_next_time_successors(successors.back(), K);
307 if (next != successors.back()) {
308 successors.push_back(next);
309 } else {
310 break;
311 }
312 }
313 return successors;
314}
