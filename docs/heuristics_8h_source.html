123template <typename ValueT, typename NodeT, typename ActionT>
124class PreferEnvironmentActionHeuristic : public Heuristic<ValueT, NodeT>
125{
126public:
130 PreferEnvironmentActionHeuristic(const std::set<ActionT> &environment_actions)
131 : environment_actions(environment_actions)
132 {
133 }
134
139 ValueT
140 compute_cost(NodeT *node) override
141 {
142 for (const auto &parent : node->parents) {
143 for (const auto &[timed_action, child] : parent->get_children()) {
144 if (child.get() == node
145 && environment_actions.find(timed_action.second) != std::end(environment_actions)) {
146 return 0;
147 }
148 }
149 }
150 return 1;
151 }
152
153private:
154 std::set<ActionT> environment_actions;
155};

177template <typename ValueT, typename NodeT>
178class CompositeHeuristic : public Heuristic<ValueT, NodeT>
179{
180public:
184 CompositeHeuristic(
185 std::vector<std::pair<ValueT, std::unique_ptr<Heuristic<ValueT, NodeT>>>> heuristics)
186 : heuristics(std::move(heuristics))
187 {
188 }
189
194 ValueT
195 compute_cost(NodeT *node) override
196 {
197 ValueT res = 0;
198 for (auto &&[weight, heuristic] : heuristics) {
199 res += weight * heuristic->compute_cost(node);
200 }
201 return res;
202 }
203
204private:
205 std::vector<std::pair<ValueT, std::unique_ptr<Heuristic<ValueT, NodeT>>>> heuristics;
206};
