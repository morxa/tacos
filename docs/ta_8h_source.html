305  TimedAutomaton(const std::set<Location> & locations,
306  const std::set<AP> & alphabet,
307  const Location & initial_location,
308  const std::set<Location> final_locations,
309  std::set<std::string> clocks,
310  const std::vector<Transition> &transitions)
311  : alphabet_(alphabet),
312  locations_(locations),
313  initial_location_(initial_location),
314  final_locations_(final_locations),
315  clocks_(clocks)
316  {
317  if (!std::includes(
318  begin(locations), end(locations), begin(final_locations), end(final_locations))) {
319  throw std::invalid_argument("Final locations must be a subset of all locations");
320  }
321  if (locations.find(initial_location) == end(locations)) {
322  throw std::invalid_argument("Initial location is not a location of the TA");
323  }
324  for (const auto &transition : transitions) {
325  for (const auto &[clock, constraint] : transition.clock_constraints_) {
326  if (clocks.find(clock) == end(clocks)) {
327  throw std::invalid_argument("Clock constraint uses unknown clock");
328  }
329  if (std::holds_alternative<AtomicClockConstraintT<std::not_equal_to<Time>>>(constraint)) {
330  throw std::invalid_argument("Inequality is not allowed in a TA guard");
331  }
332  }
333  add_transition(transition);
334  }
335  }
