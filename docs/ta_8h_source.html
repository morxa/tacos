262 TimedAutomaton(const std::set<AP> &alphabet,
263 const Location &initial_location,
264 const std::set<Location> &final_locations)
265 : alphabet_(alphabet),
266 locations_{initial_location},
267 initial_location_(initial_location),

280 TimedAutomaton(const std::set<Location> &locations,
281 const std::set<AP> &alphabet,
282 const Location &initial_location,
283 const std::set<Location> final_locations,
284 std::set<std::string> clocks,
285 const std::vector<Transition> &transitions)
286 : alphabet_(alphabet),
287 locations_(locations),
288 initial_location_(initial_location),
289 final_locations_(final_locations),
290 clocks_(clocks)
291 {
292 if (!std::includes(
293 begin(locations), end(locations), begin(final_locations), end(final_locations))) {
294 throw std::invalid_argument("Final locations must be a subset of all locations");
295 }
296 if (locations.find(initial_location) == end(locations)) {
297 throw std::invalid_argument("Initial location is not a location of the TA");
298 }
299 for (const auto &transition : transitions) {
300 for (const auto &[clock, constraint] : transition.clock_constraints_) {
301 if (clocks.find(clock) == end(clocks)) {
302 throw std::invalid_argument("Clock constraint uses unknown clock");
303 }
304 if (std::holds_alternative<AtomicClockConstraintT<std::not_equal_to<Time>>>(constraint)) {
305 throw std::invalid_argument("Inequality is not allowed in a TA guard");
306 }
307 }
308 add_transition(transition);
309 }
310 }
