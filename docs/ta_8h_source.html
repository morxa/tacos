272  TimedAutomaton(const std::set<Location> & locations,
273  const std::set<AP> & alphabet,
274  const Location & initial_location,
275  const std::set<Location> final_locations,
276  std::set<std::string> clocks,
277  const std::vector<Transition> &transitions)
278  : alphabet_(alphabet),
279  locations_(locations),
280  initial_location_(initial_location),
281  final_locations_(final_locations),
282  clocks_(clocks)
283  {
284  if (!std::includes(
285  begin(locations), end(locations), begin(final_locations), end(final_locations))) {
286  throw std::invalid_argument("Final locations must be a subset of all locations");
287  }
288  if (locations.find(initial_location) == end(locations)) {
289  throw std::invalid_argument("Initial location is not a location of the TA");
290  }
291  for (const auto &transition : transitions) {
292  for (const auto &[clock, constraint] : transition.clock_constraints_) {
293  if (clocks.find(clock) == end(clocks)) {
294  throw std::invalid_argument("Clock constraint uses unknown clock");
295  }
296  if (std::holds_alternative<AtomicClockConstraintT<std::not_equal_to<Time>>>(constraint)) {
297  throw std::invalid_argument("Inequality is not allowed in a TA guard");
298  }
299  }
300  add_transition(transition);
301  }
302  }
