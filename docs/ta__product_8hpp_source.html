27template <typename ActionT>
28std::map<ActionT, std::vector<std::size_t>>
29collect_synchronizing_alphabets(const std::set<ActionT> & synchronized_actions,
30 const std::vector<std::set<ActionT>> &alphabets)
31{
32 std::map<ActionT, std::vector<std::size_t>> res;
33 for (const auto &symbol : synchronized_actions) {
34 res[symbol] = {};
35 for (std::size_t i = 0; i < alphabets.size(); ++i) {
36 if (alphabets[i].find(symbol) != std::end(alphabets[i])) {
37 res[symbol].push_back(i);
38 }
39 }
40 }
41 return res;
42}

44template <typename LocationT, typename ActionT>
45TimedAutomaton<std::vector<LocationT>, ActionT>
46get_product(const std::vector<TimedAutomaton<LocationT, ActionT>> &automata,
47 const std::set<ActionT> & synchronized_actions)
48{
49 if (automata.empty()) {
50 throw std::invalid_argument("Cannot compute product of zero automata");
51 }
52 for (auto it1 = std::begin(automata); it1 != std::prev(std::end(automata)); ++it1) {
53 for (auto it2 = std::next(it1); it2 != std::end(automata); ++it2) {
54 std::set<std::string> common_clocks;
55 std::set_intersection(begin(it1->get_clocks()),
56 end(it1->get_clocks()),
57 begin(it2->get_clocks()),
58 end(it2->get_clocks()),
59 std::inserter(common_clocks, end(common_clocks)));
60
61 if (!common_clocks.empty()) {
62 throw std::invalid_argument(
63 fmt::format("Cannot construct product automaton for two automata with non-disjoint "
64 "clocks, common clocks: {}",
65 fmt::join(common_clocks, ", ")));
66 }
67 }
68 }
69 using ProductLocation = Location<std::vector<LocationT>>;
70 std::set<ActionT> actions;
71 std::set<ProductLocation> product_locations;
72 // Initialize the product locations with the locations from the first TA, by creating a set of
73 // Location<std::vector<LocationT>> from the set of Location<LocationT>.
74 std::transform(std::begin(std::begin(automata)->get_locations()),
75 std::end(std::begin(automata)->get_locations()),
76 std::inserter(product_locations, end(product_locations)),
77 [](const auto &location) { return ProductLocation{{location.get()}}; });
78 // Add the locations from the remaining TAs.
79 std::for_each(std::next(std::begin(automata)),
80 std::end(automata),
81 [&product_locations](const auto &ta) {
82 std::set<ProductLocation> augmented_product_locations;
83 for (const auto &product_location : product_locations) {
84 for (const auto &ta_location : ta.get_locations()) {
85 auto new_product_location = product_location;
86 new_product_location->push_back(ta_location);
87 augmented_product_locations.insert(new_product_location);
88 }
89 }
90 product_locations = augmented_product_locations;
91 });
92 std::set<ProductLocation> product_final_locations;
93 // The same as above, but for the final locations.
94 std::transform(std::begin(std::begin(automata)->get_final_locations()),
95 std::end(std::begin(automata)->get_final_locations()),
96 std::inserter(product_final_locations, end(product_final_locations)),
97 [](const auto &location) { return ProductLocation{{location.get()}}; });
98 std::for_each(std::next(std::begin(automata)),
99 std::end(automata),
100 [&product_final_locations](const auto &ta) {
101 std::set<ProductLocation> augmented_product_final_locations;
102 for (const auto &product_location : product_final_locations) {
103 for (const auto &ta_location : ta.get_final_locations()) {
104 auto new_product_location = product_location;
105 new_product_location->push_back(ta_location);
106 augmented_product_final_locations.insert(new_product_location);
107 }
108 }
109 product_final_locations = augmented_product_final_locations;
110 });
111
112 // Assert that each product location consists of one location from each TA.
113 assert(std::all_of(begin(product_locations),
114 end(product_locations),
115 [&automata](const auto &location) {
116 return location.get().size() == automata.size();
117 }));
118
119 std::set<ActionT> product_alphabet;
120 ProductLocation product_initial_location;
121 std::set<std::string> product_clocks;
122 std::vector<Transition<std::vector<LocationT>, ActionT>> product_transitions;
123 // datastructures required for synchronization
124 std::vector<std::set<ActionT>> alphabets;
125 // build alphabet, initial location and clocks of the product
126 std::for_each(std::begin(automata), std::end(automata), [&](const auto &ta) {
127 alphabets.push_back(ta.get_alphabet());
128 product_alphabet.insert(std::begin(ta.get_alphabet()), std::end(ta.get_alphabet()));
129 product_initial_location->push_back(ta.get_initial_location().get());
130 product_clocks.insert(std::begin(ta.get_clocks()), std::end(ta.get_clocks()));
131 });
132 // collect which automata synchronize on which actions
133 std::map<ActionT, std::vector<std::size_t>> synchronizing_actions =
134 collect_synchronizing_alphabets(synchronized_actions, alphabets);
135 // collects synchronizing transitions
136 std::map<ActionT, std::set<Transition<std::vector<LocationT>, ActionT>>> synchronized_transitions;
137
138 for (const auto &symbol : product_alphabet) {
139 bool symbol_synchronizes =
140 (synchronizing_actions.find(symbol) != std::end(synchronizing_actions));
141 for (const auto &[ta_i, ta] : ranges::views::enumerate(automata)) {
142 // before finalizing the construction for one automaton, we collect all candidates for
143 // synchronizing jumps to avoid expensive lookup.
144 std::map<ActionT, std::set<Transition<std::vector<LocationT>, ActionT>>>
145 new_synchronizing_jumps;
146 for (const auto &[source_location, transition] : ta.get_transitions()) {
147 // Only consider transitions where the symbol matches. Note that if the automaton does not
148 // synchronize with this symbol, this check will always fail since implicitly an automaton
149 // can only synchronize with a synchronizing symbol if it has at least one transition with
150 // this symbol. This information is relevant to establish, that the automaton is able to
151 // synchronize with this symbol in the second case (else-branch below).
152 if (transition.symbol_ == symbol) {
153 // Handling of non-synchronized transitions: insert transitions for all product locations
154 // where source and target of the local transition match.
155 if (!symbol_synchronizes) {
156 for (const auto &product_source_location : product_locations) {
157 if (product_source_location.get()[ta_i] == transition.source_.get()) {
158 auto product_target_location = product_source_location;
159 product_target_location.get()[ta_i] = transition.target_.get();
160 product_transitions.emplace_back(product_source_location,
161 transition.symbol_,
162 product_target_location,
163 transition.clock_constraints_,
164 transition.clock_resets_);
165 }
166 }
167 } else {
168 // We know the automaton *must* synchronize with the symbol
169 assert(std::find(std::begin(synchronizing_actions[symbol]),
170 std::end(synchronizing_actions[symbol]),
171 ta_i)
172 != std::end(synchronizing_actions[symbol]));
173 // if this is the first synchronized transition with this symbol, initialize set of
174 // potentially synchronizing jumps.
175 if (synchronized_transitions[symbol].empty()) {
176 for (const auto &product_source_location : product_locations) {
177 if (product_source_location.get()[ta_i] == transition.source_.get()) {
178 auto product_target_location = product_source_location;
179 product_target_location.get()[ta_i] = transition.target_.get();
180 new_synchronizing_jumps[symbol].emplace(product_source_location,
181 transition.symbol_,
182 product_target_location,
183 transition.clock_constraints_,
184 transition.clock_resets_);
185 }
186 }
187 } else {
188 // there are already transition-candidates for synchronization, we need to create
189 // candidates for each transition in the current automaton which also synchronizes.
190 for (const auto &sync_transition : synchronized_transitions[symbol]) {
191 auto product_source_location = sync_transition.source_;
192 product_source_location.get()[ta_i] = transition.source_.get();
193 auto product_target_location = sync_transition.target_;
194 product_target_location.get()[ta_i] = transition.target_.get();
195 auto constraints = sync_transition.clock_constraints_;
196 constraints.insert(std::begin(transition.clock_constraints_),
197 std::end(transition.clock_constraints_));
198 auto resets = sync_transition.clock_resets_;
199 resets.insert(std::begin(transition.clock_resets_),
200 std::end(transition.clock_resets_));
201 new_synchronizing_jumps[symbol].emplace(product_source_location,
202 transition.symbol_,
203 product_target_location,
204 constraints,
205 resets);
206 }
207 }
208 }
209 }
210 }
211 // update transitions
212 for (const auto &[symbol, transitions] : new_synchronizing_jumps) {
213 synchronized_transitions[symbol] = std::move(transitions);
214 }
215 }
216 }
217 // update transitions
218 for (const auto &[symbol, transitions] : synchronized_transitions) {
219 std::copy(std::begin(transitions),
220 std::end(transitions),
221 std::back_inserter(product_transitions));
222 }
223
224 return TimedAutomaton<std::vector<LocationT>, ActionT>{product_locations,
225 product_alphabet,
226 product_initial_location,
227 product_final_locations,
228 product_clocks,
229 product_transitions};
230}
