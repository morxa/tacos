227template <typename Location, typename ConstraintSymbolType>
228bool
229is_valid_canonical_word(const CanonicalABWord<Location, ConstraintSymbolType> &word,
230 RegionIndex max_region = 0)
231{
232 // TODO all ta_symbols should agree on the same location
233 // TODO clocks must have unique values (i.e., must not occur multiple times)
234 if (word.empty()) {
235 throw InvalidCanonicalWordException(word, "word is empty");
236 }
237 // No configuration should be empty
238 if (std::any_of(word.begin(), word.end(), [](const auto &configurations) {
239 return configurations.empty();
240 })) {
241 throw InvalidCanonicalWordException(word, "word contains an empty configuration");
242 }
243 // Each partition either contains only even or only odd region indexes. This is because the word
244 // is partitioned by the fractional part and the region index can only be even if the fractional
245 // part is 0. If that is the case, there cannot be any configuration with an odd region index in
246 // the same partition, as that configuration's fractional part would be > 0.
247 std::for_each(word.begin(), word.end(), [&word](const auto &configurations) {
248 if (std::any_of(configurations.begin(),
249 configurations.end(),
250 [](const auto &w) { return get_region_index(w) % 2 == 0; })
251 && std::any_of(configurations.begin(), configurations.end(), [](const auto &w) {
252 return get_region_index(w) % 2 == 1;
253 })) {
254 throw InvalidCanonicalWordException(word, "both odd and even region indexes");
255 }
256 });
257 // There must be no configuration with a region larger than the max region index.
258 if (max_region > 0) {
259 if (std::any_of(word.begin(), word.end(), [max_region](const auto &configurations) {
260 return std::any_of(configurations.begin(),
261 configurations.end(),
262 [max_region](const auto &w) {
263 return get_region_index(w) > max_region;
264 });
265 })) {
266 throw InvalidCanonicalWordException(
267 word, "word contains configuration with a region larger than the max region");
268 };
269 }
270 // There must be at most one partition with fractional part 0.
271 // The only partition that is allowed to have fracitonal part 0 is the 0th
272 // partition.
273 std::for_each(std::next(word.begin()), word.end(), [&word](const auto &configurations) {
274 std::for_each(configurations.begin(), configurations.end(), [&word](const auto &w) {
275 if (get_region_index(w) % 2 == 0) {
276 throw InvalidCanonicalWordException(word,
277 "fractional part 0 in wrong element of partition");
278 }
279 });
280 });
281 return true;
282}

300template <typename Location, typename ConstraintSymbolType>
301CanonicalABWord<Location, ConstraintSymbolType>
302get_canonical_word(const PlantConfiguration<Location> &plant_configuration,
303 const ATAConfiguration<ConstraintSymbolType> &ata_configuration,
304 const unsigned int K)
305{
306 using ABSymbol = ABSymbol<Location, ConstraintSymbolType>;
307 using ABRegionSymbol = ABRegionSymbol<Location, ConstraintSymbolType>;
308 // TODO Also accept a TA that does not have any clocks.
309 if (plant_configuration.clock_valuations.empty()) {
310 throw std::invalid_argument("TA without clocks are not supported");
311 }
312 std::set<ABSymbol> g;
313 // Insert ATA configurations into g.
314 std::copy(ata_configuration.begin(), ata_configuration.end(), std::inserter(g, g.end()));
315 // Insert TA configurations into g.
316 for (const auto &[clock_name, clock_value] : plant_configuration.clock_valuations) {
317 g.insert(PlantState<Location>{plant_configuration.location, clock_name, clock_value});
318 }
319 // Sort into partitions by the fractional parts.
320 std::map<ClockValuation, std::set<ABSymbol>, utilities::ApproxFloatComparator<Time>>
321 partitioned_g(utilities::ApproxFloatComparator<Time>{});
322 for (const ABSymbol &symbol : g) {
323 partitioned_g[utilities::getFractionalPart<int, ClockValuation>(get_time(symbol))].insert(
324 symbol);
325 }
326 // Replace exact clock values by region indices.
327 automata::ta::TimedAutomatonRegions regionSet{K};
328 CanonicalABWord<Location, ConstraintSymbolType> abs;
329 for (const auto &[fractional_part, g_i] : partitioned_g) {
330 std::set<ABRegionSymbol> abs_i;
331 std::transform(
332 g_i.begin(),
333 g_i.end(),
334 std::inserter(abs_i, abs_i.end()),
335 [&](const ABSymbol &w) -> ABRegionSymbol {
336 if (std::holds_alternative<PlantState<Location>>(w)) {
337 const PlantState<Location> &s = std::get<PlantState<Location>>(w);
338 return PlantRegionState<Location>{s.location,
339 s.clock,
340 regionSet.getRegionIndex(s.clock_valuation)};
341 } else {
342 const ATAState<ConstraintSymbolType> &s = std::get<ATAState<ConstraintSymbolType>>(w);
343 return ATARegionState<ConstraintSymbolType>{s.location,
344 regionSet.getRegionIndex(s.clock_valuation)};
345 }
346 });
347 abs.push_back(abs_i);
348 }
349 assert(is_valid_canonical_word(abs, 2 * K + 1));
350 return abs;
351}
