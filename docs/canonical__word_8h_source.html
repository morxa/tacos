218 template <typename Location, typename ConstraintSymbolType>
219 bool
220 is_valid_canonical_word(const CanonicalABWord<Location, ConstraintSymbolType> &word)
221 {
222  // TODO all ta_symbols should agree on the same location
223  // TODO clocks must have unique values (i.e., must not occur multiple times)
224  if (word.empty()) {
225  throw InvalidCanonicalWordException(word, "word is empty");
226  }
227  // No configuration should be empty
228  if (std::any_of(word.begin(), word.end(), [](const auto &configurations) {
229  return configurations.empty();
230  })) {
231  throw InvalidCanonicalWordException(word, "word contains an empty configuration");
232  }
233  // Each partition either contains only even or only odd region indexes. This is because the word
234  // is partitioned by the fractional part and the region index can only be even if the fractional
235  // part is 0. If that is the case, there cannot be any configuration with an odd region index in
236  // the same partition, as that configuration's fractional part would be > 0.
237  std::for_each(word.begin(), word.end(), [&word](const auto &configurations) {
238  if (std::any_of(configurations.begin(),
239  configurations.end(),
240  [](const auto &w) { return get_region_index(w) % 2 == 0; })
241  && std::any_of(configurations.begin(), configurations.end(), [](const auto &w) {
242  return get_region_index(w) % 2 == 1;
243  })) {
244  throw InvalidCanonicalWordException(word, "both odd and even region indexes");
245  }
246  });
247  // There must be at most one partition with fractional part 0.
248  // The only partition that is allowed to have fracitonal part 0 is the 0th
249  // partition.
250  std::for_each(std::next(word.begin()), word.end(), [&word](const auto &configurations) {
251  std::for_each(configurations.begin(), configurations.end(), [&word](const auto &w) {
252  if (get_region_index(w) % 2 == 0) {
253  throw InvalidCanonicalWordException(word,
254  "fractional part 0 in wrong element of partition");
255  }
256  });
257  });
258  return true;
259 }

277 template <typename Location, typename ConstraintSymbolType>
278 CanonicalABWord<Location, ConstraintSymbolType>
279 get_canonical_word(const PlantConfiguration<Location> & plant_configuration,
280  const ATAConfiguration<ConstraintSymbolType> &ata_configuration,
281  const unsigned int K)
282 {
283  using ABSymbol = ABSymbol<Location, ConstraintSymbolType>;
284  using ABRegionSymbol = ABRegionSymbol<Location, ConstraintSymbolType>;
285  // TODO Also accept a TA that does not have any clocks.
286  if (plant_configuration.clock_valuations.empty()) {
287  throw std::invalid_argument("TA without clocks are not supported");
288  }
289  std::set<ABSymbol> g;
290  // Insert ATA configurations into g.
291  std::copy(ata_configuration.begin(), ata_configuration.end(), std::inserter(g, g.end()));
292  // Insert TA configurations into g.
293  for (const auto &[clock_name, clock_value] : plant_configuration.clock_valuations) {
294  g.insert(PlantState<Location>{plant_configuration.location, clock_name, clock_value});
295  }
296  // Sort into partitions by the fractional parts.
297  std::map<ClockValuation, std::set<ABSymbol>, utilities::ApproxFloatComparator<Time>>
298  partitioned_g(utilities::ApproxFloatComparator<Time>{});
299  for (const ABSymbol &symbol : g) {
300  partitioned_g[utilities::getFractionalPart<int, ClockValuation>(get_time(symbol))].insert(
301  symbol);
302  }
303  // Replace exact clock values by region indices.
304  automata::ta::TimedAutomatonRegions regionSet{K};
305  CanonicalABWord<Location, ConstraintSymbolType> abs;
306  for (const auto &[fractional_part, g_i] : partitioned_g) {
307  std::set<ABRegionSymbol> abs_i;
308  std::transform(
309  g_i.begin(),
310  g_i.end(),
311  std::inserter(abs_i, abs_i.end()),
312  [&](const ABSymbol &w) -> ABRegionSymbol {
313  if (std::holds_alternative<PlantState<Location>>(w)) {
314  const PlantState<Location> &s = std::get<PlantState<Location>>(w);
315  return PlantRegionState<Location>{s.location,
316  s.clock,
317  regionSet.getRegionIndex(s.clock_valuation)};
318  } else {
319  const ATAState<ConstraintSymbolType> &s = std::get<ATAState<ConstraintSymbolType>>(w);
320  return ATARegionState<ConstraintSymbolType>{s.location,
321  regionSet.getRegionIndex(s.clock_valuation)};
322  }
323  });
324  abs.push_back(abs_i);
325  }
326  assert(is_valid_canonical_word(abs));
327  return abs;
328 }
