48template <typename Location, typename ActionType, typename ConstraintSymbolType>
49bool
50has_satisfiable_ata_configuration(
51 const SearchTreeNode<Location, ActionType, ConstraintSymbolType> &node)
52{
53 return !std::all_of(std::begin(node.words), std::end(node.words), [](const auto &word) {
54 return std::any_of(std::begin(word), std::end(word), [](const auto &component) {
55 return std::find_if(
56 std::begin(component),
57 std::end(component),
58 [](const auto &region_symbol) {
59 return std::holds_alternative<ATARegionState<ConstraintSymbolType>>(region_symbol)
60 && std::get<ATARegionState<ConstraintSymbolType>>(region_symbol).formula
61 == logic::MTLFormula<ConstraintSymbolType>{
62 mtl_ata_translation::get_sink<ConstraintSymbolType>()};
63 })
64 != std::end(component);
65 });
66 });
67}

71template <typename Location, typename ActionType, typename ConstraintSymbolType>
72void
73label_graph(SearchTreeNode<Location, ActionType, ConstraintSymbolType> *node,
74 const std::set<ActionType> &controller_actions,
75 const std::set<ActionType> &environment_actions,
76 std::set<SearchTreeNode<Location, ActionType, ConstraintSymbolType> *> &visited)
77{
78 if (node->label != NodeLabel::UNLABELED) {
79 return;
80 }
81 if (std::find(std::begin(visited), std::end(visited), node) != std::end(visited)) {
82 // This node was already visited, meaning that we have found a loop. In a loop, there is always
83 // a monotonic domination, because monotonic domination is reflexive.
84 node->label = NodeLabel::TOP;
85 node->label_reason = LabelReason::MONOTONIC_DOMINATION;
86 return;
87 }
88 visited.insert(node);
89 if (node->state == NodeState::GOOD) {
90 node->label_reason = LabelReason::GOOD_NODE;
91 node->set_label(NodeLabel::TOP);
92 } else if (node->state == NodeState::DEAD) {
93 node->label_reason = LabelReason::DEAD_NODE;
94 node->set_label(NodeLabel::TOP);
95 } else if (node->state == NodeState::BAD) {
96 node->label_reason = LabelReason::BAD_NODE;
97 node->set_label(NodeLabel::BOTTOM);
98 } else {
99 for (const auto &[action, child] : node->get_children()) {
100 if (child.get() != node) {
101 label_graph(child.get(), controller_actions, environment_actions, visited);
102 }
103 }
104 bool has_enviroment_step{false};
105 RegionIndex first_good_controller_step{std::numeric_limits<RegionIndex>::max()};
106 RegionIndex first_bad_environment_step{std::numeric_limits<RegionIndex>::max()};
107 for (const auto &[timed_action, child] : node->get_children()) {
108 const auto &[step, action] = timed_action;
109 if (controller_actions.find(action) != std::end(controller_actions)) {
110 assert(environment_actions.find(action) == std::end(environment_actions));
111 if (child->label == NodeLabel::TOP) {
112 first_good_controller_step = std::min(first_good_controller_step, step);
113 }
114 } else {
115 assert(environment_actions.find(action) != std::end(environment_actions));
116 has_enviroment_step = true;
117 if (child->label != NodeLabel::TOP) {
118 first_bad_environment_step = std::min(first_bad_environment_step, step);
119 }
120 }
121 }
122 // Formally, the controller selects a subset of actions U such that
123 // (1) U is deterministic: it cannot select the same action twice with different clock resets)
124 // (2) U is non-restricting: if there is an environment action at step i, then the controller
125 // must select a controller action at step j < i or it must select the environment action (3) U
126 // is non-blocking: if there is some successor, then U must not be empty. The environment then
127 // selects exactly one element of U.
128 if (first_good_controller_step < first_bad_environment_step) {
129 // The controller can just select the good controller action.
130 node->label_reason = LabelReason::GOOD_CONTROLLER_ACTION_FIRST;
131 node->set_label(NodeLabel::TOP);
132 } else if (has_enviroment_step
133 && first_bad_environment_step == std::numeric_limits<RegionIndex>::max()) {
134 // There is an environment action and no environment action is bad
135 // -> the controller can just select all environment actions
136 node->label_reason = LabelReason::NO_BAD_ENV_ACTION;
137 node->set_label(NodeLabel::TOP);
138 } else if (!has_enviroment_step) {
139 // All controller actions must be bad (otherwise we would be in the first case)
140 // -> no controller strategy
141 assert(first_good_controller_step == std::numeric_limits<RegionIndex>::max());
142 node->label_reason = LabelReason::ALL_CONTROLLER_ACTIONS_BAD;
143 node->set_label(NodeLabel::BOTTOM);
144 } else {
145 // There must be an environment action (otherwise case 3) and one of them must be bad
146 // (otherwise case 2).
147 assert(first_bad_environment_step < std::numeric_limits<RegionIndex>::max());
148 node->label_reason = LabelReason::BAD_ENV_ACTION_FIRST;
149 node->set_label(NodeLabel::BOTTOM);
150 }
151 }
152}

178template <typename Location,
179 typename ActionType,
180 typename ConstraintSymbolType = ActionType,
181 bool use_location_constraints = false,
182 typename Plant =
183 automata::ta::TimedAutomaton<typename Location::UnderlyingType, ActionType>,
184 bool use_set_semantics = false>
185class TreeSearch
186{
187public:
189 using ATAInputType =
190 /* if use_set_semantics is true, then the ATASymbolT is the same as
191 * std::set<ConstraintSymbolType>, otherwise it is just ConstraintSymbolType> */
192 typename std::disjunction<
193 utilities::values_equal<use_set_semantics, false, ConstraintSymbolType>,
194 utilities::values_equal<use_set_semantics, true, std::set<ConstraintSymbolType>>,
195 std::void_t<void>>::type;
197 using Node = SearchTreeNode<Location, ActionType, ConstraintSymbolType>;
198
209 TreeSearch(
210 // const automata::ta::TimedAutomaton<Location, ActionType> * ta,
211 const Plant *ta,
212 automata::ata::AlternatingTimedAutomaton<logic::MTLFormula<ConstraintSymbolType>,
213 logic::AtomicProposition<ATAInputType>> *ata,
214 std::set<ActionType> controller_actions,
215 std::set<ActionType> environment_actions,
216 RegionIndex K,
217 bool incremental_labeling = false,
218 bool terminate_early = false,
219 std::unique_ptr<Heuristic<long, Node>> search_heuristic =
220 std::make_unique<BfsHeuristic<long, Node>>())
221 : ta_(ta),
222 ata_(ata),
223 controller_actions_(controller_actions),
224 environment_actions_(environment_actions),
225 K_(K),
226 incremental_labeling_(incremental_labeling),
227 terminate_early_(terminate_early)
228 {
229 static_assert(use_location_constraints || std::is_same_v<ActionType, ConstraintSymbolType>);
230 // Assert that the two action sets are disjoint.
231 assert(
232 std::all_of(controller_actions_.begin(), controller_actions_.end(), [this](const auto &a) {
233 return environment_actions_.find(a) == environment_actions_.end();
234 }));
235 assert(
236 std::all_of(environment_actions_.begin(), environment_actions_.end(), [this](const auto &a) {
237 return controller_actions_.find(a) == controller_actions_.end();
238 }));
239 if constexpr (use_location_constraints && use_set_semantics) {
240 // Already read a single empty symbol to skip l0.
241 const auto ata_configurations =
242 ata->make_symbol_step(ata->get_initial_configuration(),
243 logic::AtomicProposition<ATAInputType>{{}});
244 std::set<CanonicalABWord<typename Plant::Location, ConstraintSymbolType>> initial_words;
245 for (const auto &ata_configuration : ata_configurations) {
246 initial_words.insert(
247 get_canonical_word(ta->get_initial_configuration(), ata_configuration, K));
248 }
249
250 tree_root_ = std::make_shared<Node>(initial_words);
251 } else {
252 tree_root_ = std::make_shared<Node>(
253 std::set<CanonicalABWord<typename Plant::Location, ConstraintSymbolType>>{
254 get_canonical_word(ta->get_initial_configuration(),
255 ata->get_initial_configuration(),
256 K)});
257 }
258 nodes_ = {{{}, tree_root_}};
259 heuristic = std::move(search_heuristic);
260 tree_root_->min_total_region_increments = 0;
261 add_node_to_queue(tree_root_.get());
262 }

314 bool
315 step()
316 {
317 utilities::QueueAccess queue_access{&pool_};
318 SPDLOG_TRACE("Getting next node from queue, queue size is {}", queue_access.get_size());
319 if (queue_access.empty()) {
320 return false;
321 }
322 auto step_function = std::get<1>(queue_access.top());
323 queue_access.pop();
324 step_function();
325 return true;
326 }

329 void
330 expand_node(Node *node)
331 {
332 if (node->label != NodeLabel::UNLABELED) {
333 // The node was already labeled, nothing to do.
334 return;
335 }
336 bool is_expanding = node->is_expanding.exchange(true);
337 if (is_expanding) {
338 // The node is already being expanded.
339 return;
340 }
341 SPDLOG_TRACE("Processing {}", *node);
342 if (is_bad_node(node)) {
343 SPDLOG_DEBUG("Node {} is BAD", *node);
344 node->label_reason = LabelReason::BAD_NODE;
345 node->state = NodeState::BAD;
346 node->is_expanded = true;
347 node->is_expanding = false;
348 if (incremental_labeling_) {
349 node->set_label(NodeLabel::BOTTOM, terminate_early_);
350 node->label_propagate(controller_actions_, environment_actions_, terminate_early_);
351 }
352 return;
353 }
354 if (!has_satisfiable_ata_configuration(*node)) {
355 node->label_reason = LabelReason::NO_ATA_SUCCESSOR;
356 node->state = NodeState::GOOD;
357 node->is_expanded = true;
358 node->is_expanding = false;
359 if (incremental_labeling_) {
360 node->set_label(NodeLabel::TOP, terminate_early_);
361 node->label_propagate(controller_actions_, environment_actions_, terminate_early_);
362 }
363 return;
364 }
365 if (dominates_ancestor(node)) {
366 node->label_reason = LabelReason::MONOTONIC_DOMINATION;
367 node->state = NodeState::GOOD;
368 node->is_expanded = true;
369 node->is_expanding = false;
370 if (incremental_labeling_) {
371 node->set_label(NodeLabel::TOP, terminate_early_);
372 node->label_propagate(controller_actions_, environment_actions_, terminate_early_);
373 }
374 return;
375 }
376
377 std::set<Node *> new_children;
378 std::set<Node *> existing_children;
379 if (node->get_children().empty()) {
380 std::tie(new_children, existing_children) = compute_children(node);
381 }
382
383 node->is_expanded = true;
384 node->is_expanding = false;
385 if (node->label == NodeLabel::CANCELED) {
386 // The node has been canceled in the meantime, do not add children to queue.
387 return;
388 }
389 for (const auto &child : existing_children) {
390 SPDLOG_TRACE("Found existing node for {}", fmt::ptr(child));
391 if (child->label == NodeLabel::CANCELED) {
392 SPDLOG_DEBUG("Expansion of {}: Found existing child {}, is canceled, re-adding",
393 fmt::ptr(node),
394 fmt::ptr(child));
395 child->reset_label();
396 add_node_to_queue(child);
397 }
398 }
399 if (incremental_labeling_ && !existing_children.empty()) {
400 // There is an existing child, directly check the labeling.
401 SPDLOG_TRACE("Node {} has existing child, updating labels", node_to_string(*node, false));
402 node->label_propagate(controller_actions_, environment_actions_, terminate_early_);
403 }
404 for (const auto &child : new_children) {
405 add_node_to_queue(child);
406 }
407 SPDLOG_TRACE("Node has {} children, {} of them new",
408 node->get_children().size(),
409 new_children.size());
410 if (node->get_children().empty()) {
411 node->label_reason = LabelReason::DEAD_NODE;
412 node->state = NodeState::DEAD;
413 if (incremental_labeling_) {
414 node->set_label(NodeLabel::TOP, terminate_early_);
415 node->label_propagate(controller_actions_, environment_actions_, terminate_early_);
416 }
417 }
418 }

449private:
450 std::pair<std::set<Node *>, std::set<Node *>>
451 compute_children(Node *node)
452 {
453 if (node == nullptr) {
454 return {};
455 }
456 assert(node->get_children().empty());
457 std::map<std::pair<RegionIndex, ActionType>,
458 std::set<CanonicalABWord<Location, ConstraintSymbolType>>>
459 child_classes;
460
461 const auto time_successors = get_time_successors(node->words, K_);
462 for (std::size_t increment = 0; increment < time_successors.size(); ++increment) {
463 for (const auto &time_successor : time_successors[increment]) {
464 auto successors =
465 get_next_canonical_words<Plant,
466 ActionType,
467 ConstraintSymbolType,
468 use_location_constraints,
469 use_set_semantics>(controller_actions_, environment_actions_)(
470 *ta_, *ata_, get_candidate(time_successor), increment, K_);
471 for (const auto &[symbol, successor] : successors) {
472 assert(
473 std::find(std::begin(controller_actions_), std::end(controller_actions_), symbol)
474 != std::end(controller_actions_)
475 || std::find(std::begin(environment_actions_), std::end(environment_actions_), symbol)
476 != std::end(environment_actions_));
477 child_classes[std::make_pair(increment, symbol)].insert(successor);
478 }
479 }
480 }
481
482 std::set<Node *> new_children;
483 std::set<Node *> existing_children;
484 // Create child nodes, where each child contains all successors words of
485 // the same reg_a class.
486 {
487 std::lock_guard lock{nodes_mutex_};
488 for (const auto &[timed_action, words] : child_classes) {
489 auto [child_it, is_new] = nodes_.insert({words, std::make_shared<Node>(words)});
490 const std::shared_ptr<Node> &child_ptr = child_it->second;
491 node->add_child(timed_action, child_ptr);
492 SPDLOG_TRACE("Action ({}, {}): Adding child {}",
493 timed_action.first,
494 timed_action.second,
495 words);
496 if (is_new) {
497 new_children.insert(child_ptr.get());
498 } else {
499 existing_children.insert(child_ptr.get());
500 }
501 }
502 }
503 return {new_children, existing_children};
504 }
