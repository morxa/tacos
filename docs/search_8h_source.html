52 template <typename Location, typename ActionType, typename ConstraintSymbolType>
53 bool
54 has_satisfiable_ata_configuration(
55  const SearchTreeNode<Location, ActionType, ConstraintSymbolType> &node)
56 {
57  return !std::all_of(std::begin(node.words), std::end(node.words), [](const auto &word) {
58  return std::any_of(std::begin(word), std::end(word), [](const auto &component) {
59  return std::find_if(
60  std::begin(component),
61  std::end(component),
62  [](const auto &region_symbol) {
63  return std::holds_alternative<ATARegionState<ConstraintSymbolType>>(region_symbol)
64  && std::get<ATARegionState<ConstraintSymbolType>>(region_symbol).formula
65  == logic::MTLFormula<ConstraintSymbolType>{
66  mtl_ata_translation::get_sink<ConstraintSymbolType>()};
67  })
68  != std::end(component);
69  });
70  });
71 }

185  bool
186  step()
187  {
188  utilities::QueueAccess queue_access{&pool_};
189  SPDLOG_TRACE("Getting next node from queue, queue size is {}", queue_access.get_size());
190  if (queue_access.empty()) {
191  return false;
192  }
193  auto step_function = std::get<1>(queue_access.top());
194  queue_access.pop();
195  step_function();
196  return true;
197  }

200  void
201  expand_node(Node *node)
202  {
203  if (node->label != NodeLabel::UNLABELED) {
204  // The node was already labeled, nothing to do.
205  return;
206  }
207  bool is_expanding = node->is_expanding.exchange(true);
208  if (is_expanding) {
209  // The node is already being expanded.
210  return;
211  }
212  SPDLOG_TRACE("Processing {}", *node);
213  if (is_bad_node(node)) {
214  SPDLOG_DEBUG("Node {} is BAD", *node);
215  node->label_reason = LabelReason::BAD_NODE;
216  node->state = NodeState::BAD;
217  node->is_expanded = true;
218  node->is_expanding = false;
219  if (incremental_labeling_) {
220  node->set_label(NodeLabel::BOTTOM, terminate_early_);
221  node->label_propagate(controller_actions_, environment_actions_, terminate_early_);
222  }
223  return;
224  }
225  if (!has_satisfiable_ata_configuration(*node)) {
226  node->label_reason = LabelReason::NO_ATA_SUCCESSOR;
227  node->state = NodeState::GOOD;
228  node->is_expanded = true;
229  node->is_expanding = false;
230  if (incremental_labeling_) {
231  node->set_label(NodeLabel::TOP, terminate_early_);
232  node->label_propagate(controller_actions_, environment_actions_, terminate_early_);
233  }
234  return;
235  }
236  if (dominates_ancestor(node)) {
237  node->label_reason = LabelReason::MONOTONIC_DOMINATION;
238  node->state = NodeState::GOOD;
239  node->is_expanded = true;
240  node->is_expanding = false;
241  if (incremental_labeling_) {
242  node->set_label(NodeLabel::TOP, terminate_early_);
243  node->label_propagate(controller_actions_, environment_actions_, terminate_early_);
244  }
245  return;
246  }
247 
248  std::set<Node *> new_children;
249  std::set<Node *> existing_children;
250  if (node->get_children().empty()) {
251  std::tie(new_children, existing_children) = compute_children(node);
252  }
253 
254  node->is_expanded = true;
255  node->is_expanding = false;
256  if (node->label == NodeLabel::CANCELED) {
257  // The node has been canceled in the meantime, do not add children to queue.
258  return;
259  }
260  for (const auto &child : existing_children) {
261  SPDLOG_TRACE("Found existing node for {}", fmt::ptr(child));
262  if (child->label == NodeLabel::CANCELED) {
263  SPDLOG_DEBUG("Expansion of {}: Found existing child {}, is canceled, re-adding",
264  fmt::ptr(node),
265  fmt::ptr(child));
266  child->reset_label();
267  add_node_to_queue(child);
268  }
269  }
270  if (incremental_labeling_ && !existing_children.empty()) {
271  // There is an existing child, directly check the labeling.
272  SPDLOG_TRACE("Node {} has existing child, updating labels", node_to_string(*node, false));
273  node->label_propagate(controller_actions_, environment_actions_, terminate_early_);
274  }
275  for (const auto &child : new_children) {
276  add_node_to_queue(child);
277  }
278  SPDLOG_TRACE("Node has {} children, {} of them new",
279  node->get_children().size(),
280  new_children.size());
281  if (node->get_children().empty()) {
282  node->label_reason = LabelReason::DEAD_NODE;
283  node->state = NodeState::DEAD;
284  if (incremental_labeling_) {
285  node->set_label(NodeLabel::TOP, terminate_early_);
286  node->label_propagate(controller_actions_, environment_actions_, terminate_early_);
287  }
288  }
289  }

294  void
295  label(Node *node = nullptr)
296  {
297  // TODO test the label function separately.
298  if (node == nullptr) {
299  node = get_root();
300  }
301  if (node->label != NodeLabel::UNLABELED) {
302  return;
303  }
304  if (node->state == NodeState::GOOD) {
305  node->label_reason = LabelReason::GOOD_NODE;
306  node->set_label(NodeLabel::TOP, terminate_early_);
307  } else if (node->state == NodeState::DEAD) {
308  node->label_reason = LabelReason::DEAD_NODE;
309  node->set_label(NodeLabel::TOP, terminate_early_);
310  } else if (node->state == NodeState::BAD) {
311  node->label_reason = LabelReason::BAD_NODE;
312  node->set_label(NodeLabel::BOTTOM, terminate_early_);
313  } else {
314  for (const auto &[action, child] : node->get_children()) {
315  if (child.get() != node) {
316  label(child.get());
317  }
318  }
319  bool found_bad = false;
320  RegionIndex first_good_controller_step{std::numeric_limits<RegionIndex>::max()};
321  RegionIndex first_bad_environment_step{std::numeric_limits<RegionIndex>::max()};
322  for (const auto &[timed_action, child] : node->get_children()) {
323  const auto &[step, action] = timed_action;
324  if (child->label == NodeLabel::TOP
325  && controller_actions_.find(action) != std::end(controller_actions_)) {
326  first_good_controller_step = std::min(first_good_controller_step, step);
327  } else if (child->label == NodeLabel::BOTTOM
328  && environment_actions_.find(action) != std::end(environment_actions_)) {
329  found_bad = true;
330  first_bad_environment_step = std::min(first_bad_environment_step, step);
331  }
332  }
333  if (!found_bad) {
334  node->label_reason = LabelReason::NO_BAD_ENV_ACTION;
335  node->set_label(NodeLabel::TOP, terminate_early_);
336  } else if (first_good_controller_step < first_bad_environment_step) {
337  node->label_reason = LabelReason::GOOD_CONTROLLER_ACTION_FIRST;
338  node->set_label(NodeLabel::TOP, terminate_early_);
339  } else {
340  node->label_reason = LabelReason::BAD_ENV_ACTION_FIRST;
341  node->set_label(NodeLabel::BOTTOM, terminate_early_);
342  }
343  }
344  }

364  std::pair<std::set<Node *>, std::set<Node *>>
365  compute_children(Node *node)
366  {
367  if (node == nullptr) {
368  return {};
369  }
370  assert(node->get_children().empty());
371  // Represent a set of configurations by their reg_a component so we can later partition the
372  // set
373  std::map<std::pair<RegionIndex, ActionType>,
374  std::map<CanonicalABWord<Location, ConstraintSymbolType>,
375  std::set<CanonicalABWord<Location, ConstraintSymbolType>>>>
376  child_classes;
377 
378  for (const auto &word : node->words) {
379  for (const auto &[increment, time_successor] : get_time_successors(word, K_)) {
380  auto successors = get_next_canonical_words<Plant,
381  ActionType,
382  ConstraintSymbolType,
383  use_location_constraints>(controller_actions_,
384  environment_actions_)(
385  *ta_, *ata_, get_candidate(time_successor), increment, K_);
386  for (const auto &[symbol, successor] : successors) {
387  const auto word_reg = reg_a(successor);
388  child_classes[std::make_pair(increment, symbol)][word_reg].insert(successor);
389  }
390  }
391  }
392 
393  std::set<Node *> new_children;
394  std::set<Node *> existing_children;
395  // Create child nodes, where each child contains all successors words of
396  // the same reg_a class.
397  {
398  std::lock_guard lock{nodes_mutex_};
399  for (const auto &[timed_action, word_map] : child_classes) {
400  for (const auto &[reg_a, words] : word_map) {
401  auto [child_it, is_new] = nodes_.insert({words, std::make_shared<Node>(words)});
402  const std::shared_ptr<Node> &child_ptr = child_it->second;
403  node->add_child(timed_action, child_ptr);
404  if (is_new) {
405  SPDLOG_TRACE("New child: {}", words);
406  new_children.insert(child_ptr.get());
407  } else {
408  existing_children.insert(child_ptr.get());
409  }
410  }
411  }
412  }
413  return {new_children, existing_children};
414  }
