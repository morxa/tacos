72template <typename LocationT, typename AP>
73bool
74Transition<LocationT, AP>::is_enabled(const AP &symbol, const ClockSetValuation &clock_vals) const
75{
76 if (symbol != symbol_) {
77 return false;
78 }
79 return std::all_of(std::begin(clock_constraints_),
80 std::end(clock_constraints_),
81 [&clock_vals](const auto &constraint) {
82 return is_satisfied(constraint.second, clock_vals.at(constraint.first));
83 });
84}

95template <typename LocationT, typename AP>
96bool
97operator<(const Transition<LocationT, AP> &lhs, const Transition<LocationT, AP> &rhs)
98{
99 // cheap checks first
100 auto left_tie = std::tie(lhs.source_, lhs.target_, lhs.symbol_, lhs.clock_resets_);
101 auto right_tie = std::tie(rhs.source_, rhs.target_, rhs.symbol_, rhs.clock_resets_);
102 if (left_tie != right_tie) {
103 return left_tie < right_tie;
104 }
105 // compare clock constraints
106 auto lhs_clocks_it = std::begin(lhs.clock_constraints_);
107 auto rhs_clocks_it = std::begin(rhs.clock_constraints_);
108 while (lhs_clocks_it != std::end(lhs.clock_constraints_)
109 && rhs_clocks_it != std::end(rhs.clock_constraints_)) {
110 // compare which clocks are constrained
111 if (lhs_clocks_it->first == rhs_clocks_it->first) {
112 // compare the constraints on a single clock, if clocks are equal
113 if (lhs_clocks_it->second == rhs_clocks_it->second) {
114 ++lhs_clocks_it;
115 ++rhs_clocks_it;
116 continue;
117 } else {
118 return lhs_clocks_it->second < rhs_clocks_it->second;
119 }
120 } else {
121 return lhs_clocks_it->first < rhs_clocks_it->first;
122 }
123 ++lhs_clocks_it;
124 ++rhs_clocks_it;
125 }
126 if (lhs_clocks_it == std::end(lhs.clock_constraints_)
127 && rhs_clocks_it != std::end(rhs.clock_constraints_)) {
128 return true;
129 } else if (lhs_clocks_it != std::end(lhs.clock_constraints_)
130 && rhs_clocks_it == std::end(rhs.clock_constraints_)) {
131 return false;
132 }
133 // both are equal
134 return false;
135}

154template <typename LocationT, typename AP>
155void
156TimedAutomaton<LocationT, AP>::add_transition(const Transition &transition)
157{
158 if (alphabet_.count(transition.symbol_) == 0) {
159 throw InvalidSymbolException(transition.symbol_);
160 }
161 if (!locations_.count(transition.source_)) {
162 throw InvalidLocationException(transition.source_);
163 }
164 if (!locations_.count(transition.target_)) {
165 throw InvalidLocationException(transition.target_);
166 }
167 for (const auto &[clock_name, constraint] : transition.clock_constraints_) {
168 if (!clocks_.count(clock_name)) {
169 throw InvalidClockException(clock_name);
170 };
171 }
172 for (const auto &clock_name : transition.clock_resets_) {
173 if (!clocks_.count(clock_name)) {
174 throw InvalidClockException(clock_name);
175 };
176 }
177 transitions_.insert({transition.source_, transition});
178}

180template <typename LocationT, typename AP>
181std::set<TAConfiguration<LocationT>>
182TimedAutomaton<LocationT, AP>::make_symbol_step(const TAConfiguration<LocationT> &configuration,
183 const AP &symbol) const
184{
185 std::set<TAConfiguration<LocationT>> res;
186 // TODO This may cause an issue if the transitions are not sorted as expected, because
187 // equal_range returns a sub-sequence
188 auto [first, last] = transitions_.equal_range(configuration.location);
189 while (first != last) {
190 auto trans = std::find_if(first, last, [&](const auto &trans) {
191 return trans.second.is_enabled(symbol, configuration.clock_valuations);
192 });
193 if (trans == last) {
194 return res;
195 }
196 first = std::next(trans);
197 ClockSetValuation next_clocks = configuration.clock_valuations;
198 for (const auto &name : trans->second.clock_resets_) {
199 next_clocks[name].reset();
200 }
201 res.insert(TAConfiguration<LocationT>{trans->second.target_, next_clocks});
202 }
203 return res;
204}

206template <typename LocationT, typename AP>
207std::set<Path<LocationT, AP>>
208TimedAutomaton<LocationT, AP>::make_transition(Path<LocationT, AP> path,
209 const AP &symbol,
210 const Time &time) const
211{
212 if (path.tick_ > time) {
213 return {};
214 }
215 for (auto &[name, clock] : path.clock_valuations_) {
216 clock.tick(time - path.tick_);
217 }
218 path.tick_ = time;
219 std::set<Path<LocationT, AP>> paths;
220 TAConfiguration<LocationT> start_configuration = path.get_current_configuration();
221 for (const auto &target_configuration : make_symbol_step(start_configuration, symbol)) {
222 auto new_path = path;
223 path.sequence_.emplace_back(symbol, time, target_configuration.location);
224 new_path.current_location_ = target_configuration.location;
225 new_path.clock_valuations_ = target_configuration.clock_valuations;
226 paths.insert(new_path);
227 }
228 return paths;
229}

231template <typename LocationT, typename AP>
232bool
233TimedAutomaton<LocationT, AP>::accepts_word(const TimedWord &word) const
234{
235 std::set<Path<LocationT, AP>> paths{Path<LocationT, AP>{initial_location_, clocks_}};
236 for (auto &[symbol, time] : word) {
237 std::set<Path<LocationT, AP>> res_paths;
238 for (auto &path : paths) {
239 auto new_paths = make_transition(path, symbol, time);
240 res_paths.insert(std::begin(new_paths), std::end(new_paths));
241 }
242 paths = res_paths;
243 if (paths.empty()) {
244 return false;
245 }
246 }
247 for (auto &path : paths) {
248 if (final_locations_.find(path.current_location_) != final_locations_.end()) {
249 return true;
250 };
251 }
252 return false;
253}

256std::vector<Transition<LocationT, AP>>
257TimedAutomaton<LocationT, AP>::get_enabled_transitions(
258 const TAConfiguration<LocationT> &configuration)
259{
260 std::vector<Transition> res;
261 for (const auto &symbol : alphabet_) {
262 for (const auto &[source, transition] : transitions_) {
263 if (source == configuration.location

265 res.push_back(transition);
266 }
267 }
268 }
269 return res;

272template <typename LocationT, typename AP>
273Endpoint
274TimedAutomaton<LocationT, AP>::get_largest_constant() const
275{
276 Endpoint res{0};
277 for (const auto &[symbol, transition] : transitions_) {
278 for (const auto &[symbol, constraint] : transition.get_guards()) {
279 const auto candidate =
280 std::visit([](const auto &c) { return c.get_comparand(); }, constraint);
281 if (candidate > res) {
282 res = candidate;
283 }
284 }
285 }
286 return res;
287}
