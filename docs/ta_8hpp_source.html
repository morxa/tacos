95 template <typename LocationT, typename AP>
96 bool
97 operator<(const Transition<LocationT, AP> &lhs, const Transition<LocationT, AP> &rhs)
98 {
99  // cheap checks first
100  auto left_tie = std::tie(lhs.source_, lhs.target_, lhs.symbol_, lhs.clock_resets_);
101  auto right_tie = std::tie(rhs.source_, rhs.target_, rhs.symbol_, rhs.clock_resets_);
102  if (left_tie != right_tie) {
103  return left_tie < right_tie;
104  }
105  // compare clock constraints
106  auto lhs_clocks_it = std::begin(lhs.clock_constraints_);
107  auto rhs_clocks_it = std::begin(rhs.clock_constraints_);
108  while (lhs_clocks_it != std::end(lhs.clock_constraints_)
109  && rhs_clocks_it != std::end(rhs.clock_constraints_)) {
110  // compare which clocks are constrained
111  if (lhs_clocks_it->first == rhs_clocks_it->first) {
112  // compare the constraints on a single clock, if clocks are equal
113  if (lhs_clocks_it->second == rhs_clocks_it->second) {
114  ++lhs_clocks_it;
115  ++rhs_clocks_it;
116  continue;
117  } else {
118  return lhs_clocks_it->second < rhs_clocks_it->second;
119  }
120  } else {
121  return lhs_clocks_it->first < rhs_clocks_it->first;
122  }
123  ++lhs_clocks_it;
124  ++rhs_clocks_it;
125  }
126  if (lhs_clocks_it == std::end(lhs.clock_constraints_)
127  && rhs_clocks_it != std::end(rhs.clock_constraints_)) {
128  return true;
129  } else if (lhs_clocks_it != std::end(lhs.clock_constraints_)
130  && rhs_clocks_it == std::end(rhs.clock_constraints_)) {
131  return false;
132  }
133  // both are equal
134  return false;
135 }

153 template <typename LocationT, typename AP>
154 void
155 TimedAutomaton<LocationT, AP>::add_transition(const Transition &transition)
156 {
157  if (alphabet_.count(transition.symbol_) == 0) {
158  throw InvalidSymbolException(transition.symbol_);
159  }
160  if (!locations_.count(transition.source_)) {
161  throw InvalidLocationException(transition.source_);
162  }
163  if (!locations_.count(transition.target_)) {
164  throw InvalidLocationException(transition.target_);
165  }
166  for (const auto &[clock_name, constraint] : transition.clock_constraints_) {
167  if (!clocks_.count(clock_name)) {
168  throw InvalidClockException(clock_name);
169  };
170  }
171  for (const auto &clock_name : transition.clock_resets_) {
172  if (!clocks_.count(clock_name)) {
173  throw InvalidClockException(clock_name);
174  };
175  }
176  transitions_.insert({transition.source_, transition});
177 }

179 template <typename LocationT, typename AP>
180 std::set<TAConfiguration<LocationT>>
181 TimedAutomaton<LocationT, AP>::make_symbol_step(const TAConfiguration<LocationT> &configuration,
182  const AP & symbol) const
183 {
184  std::set<TAConfiguration<LocationT>> res;
185  // TODO This may cause an issue if the transitions are not sorted as expected, because
186  // equal_range returns a sub-sequence
187  auto [first, last] = transitions_.equal_range(configuration.location);
188  while (first != last) {
189  auto trans = std::find_if(first, last, [&](const auto &trans) {
190  return trans.second.is_enabled(symbol, configuration.clock_valuations);
191  });
192  if (trans == last) {
193  return res;
194  }
195  first = std::next(trans);
196  ClockSetValuation next_clocks = configuration.clock_valuations;
197  for (const auto &name : trans->second.clock_resets_) {
198  next_clocks[name].reset();
199  }
200  res.insert(TAConfiguration<LocationT>{trans->second.target_, next_clocks});
201  }
202  return res;
203 }

205 template <typename LocationT, typename AP>
206 std::set<Path<LocationT, AP>>
207 TimedAutomaton<LocationT, AP>::make_transition(Path<LocationT, AP> path,
208  const AP & symbol,
209  const Time & time) const
210 {
211  if (path.tick_ > time) {
212  return {};
213  }
214  for (auto &[name, clock] : path.clock_valuations_) {
215  clock.tick(time - path.tick_);
216  }
217  path.tick_ = time;
218  std::set<Path<LocationT, AP>> paths;
219  TAConfiguration<LocationT> start_configuration = path.get_current_configuration();
220  for (const auto &target_configuration : make_symbol_step(start_configuration, symbol)) {
221  auto new_path = path;
222  path.sequence_.emplace_back(symbol, time, target_configuration.location);
223  new_path.current_location_ = target_configuration.location;
224  new_path.clock_valuations_ = target_configuration.clock_valuations;
225  paths.insert(new_path);
226  }
227  return paths;
228 }

230 template <typename LocationT, typename AP>
231 bool
232 TimedAutomaton<LocationT, AP>::accepts_word(const TimedWord &word) const
233 {
234  std::set<Path<LocationT, AP>> paths{Path<LocationT, AP>{initial_location_, clocks_}};
235  for (auto &[symbol, time] : word) {
236  std::set<Path<LocationT, AP>> res_paths;
237  for (auto &path : paths) {
238  auto new_paths = make_transition(path, symbol, time);
239  res_paths.insert(std::begin(new_paths), std::end(new_paths));
240  }
241  paths = res_paths;
242  if (paths.empty()) {
243  return false;
244  }
245  }
246  for (auto &path : paths) {
247  if (final_locations_.find(path.current_location_) != final_locations_.end()) {
248  return true;
249  };
250  }
251  return false;
252 }

274 {
