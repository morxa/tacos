34template <typename ActionT, typename NodeT>
35std::map<int, const NodeT *>
36create_selector_map(
37 const std::map<std::pair<RegionIndex, ActionT>, std::shared_ptr<NodeT>> &children,
38 const std::set<NodeT *> &parents = {})
39{
40 std::map<int, const NodeT *> selector_map;
41 int node_index = 0;
42 for (const auto &node : parents) {
43 selector_map[node_index] = node;
44 fmt::print("{}: Parent \033[37m{}\033[0m\n", node_index, *node);
45 node_index += 1;
46 }
47 for (const auto &[action, node] : children) {
48 selector_map[node_index] = node.get();
49 fmt::print("{}: \033[34m({}, {})\033[0m -> \033[37m{}\033[0m\n",
50 node_index,
51 action.first,
52 action.second,
53 *node);
54 node_index += 1;
55 }
56 return selector_map;
57}

69template <typename LocationT, typename ActionT, typename ConstraintSymbolT>
70void
71search_tree_to_graphviz_interactive(
72 const search::SearchTreeNode<LocationT, ActionT, ConstraintSymbolT> *search_node,
73 const std::filesystem::path &output_path,
74 std::istream &is = std::cin)
75{
76 using Node = search::SearchTreeNode<LocationT, ActionT, ConstraintSymbolT>;
77 std::vector<const Node *> selected_nodes = {search_node};
78 const Node *last_node = search_node;
79 std::function<bool(const Node &)> selector = [&selected_nodes](const Node &node) {
80 return std::find(std::begin(selected_nodes), std::end(selected_nodes), &node)
81 != std::end(selected_nodes);
82 };
83 bool quit = false;
84 Mode mode = Mode::INSERT;
85 fmt::print("Starting interactive debugger\n");
86 print_interactive_help();
87 while (!quit) {
88 fmt::print("Updating output file {} ...", output_path);
89 search_tree_to_graphviz(*search_node, selector).render_to_file(output_path);
90 fmt::print(" done!\n");
91 fmt::print("{}: Please select a child (or 'h' for help):\n", mode);
92
93 std::map<int, const Node *> selector_map;
94 if (mode == Mode::NAVIGATE) {
95 selector_map =
96 details::create_selector_map<ActionT, Node>(last_node->get_children(), last_node->parents);
97 } else {
98 selector_map = details::create_selector_map<ActionT, Node>(last_node->get_children());
99 }
100
101 std::string input;
102 std::getline(is, input);
103 if (input == "q") {
104 quit = true;
105 } else if (input == "h") {
106 print_interactive_help();
107 } else if (input == "u") {
108 if (selected_nodes.size() <= 1) {
109 fmt::print("Cannot remove the last node!\n");
110 } else {
111 selected_nodes.pop_back();
112 last_node = selected_nodes.back();
113 }
114 } else if (input == "n") {
115 mode = Mode::NAVIGATE;
116 } else if (input == "i") {
117 mode = Mode::INSERT;
118 } else if (input == "a") {
119 mode = Mode::INSERT_AND_FOLLOW;
120 } else if (mode == Mode::INSERT && input == "*") {
121 for (const auto &[_, child] : selector_map) {
122 selected_nodes.push_back(child);
123 }
124 } else {
125 std::size_t selected = 0;
126 try {
127 selected = std::stoi(input);
128 } catch (std::exception &e) {
129 fmt::print("Failed to parse input '{}': {}\n", input, e.what());
130 continue;
131 }
132 if (selected >= selector_map.size()) {
133 fmt::print("Invalid input (must be in range [0, {}))\n", last_node->get_children().size());
134 continue;
135 }
136 auto &selected_node = selector_map.at(selected);
137 switch (mode) {
138 case Mode::NAVIGATE: last_node = selected_node; break;
139 case Mode::INSERT: selected_nodes.push_back(selected_node); break;
140 case Mode::INSERT_AND_FOLLOW:
141 last_node = selected_node;
142 selected_nodes.push_back(selected_node);
143 break;
144 }
145 }
146 }
147}
