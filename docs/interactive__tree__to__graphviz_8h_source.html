33 template <typename ActionT, typename NodeT>
34 std::map<int, const NodeT *>
35 create_selector_map(
36  const std::map<std::pair<RegionIndex, ActionT>, std::shared_ptr<NodeT>> &children,
37  const std::set<NodeT *> &parents = {})
38 {
39  std::map<int, const NodeT *> selector_map;
40  int node_index = 0;
41  for (const auto &node : parents) {
42  selector_map[node_index] = node;
43  fmt::print("{}: Parent \033[37m{}\033[0m\n", node_index, *node);
44  node_index += 1;
45  }
46  for (const auto &[action, node] : children) {
47  selector_map[node_index] = node.get();
48  fmt::print("{}: \033[34m({}, {})\033[0m -> \033[37m{}\033[0m\n",
49  node_index,
50  action.first,
51  action.second,
52  *node);
53  node_index += 1;
54  }
55  return selector_map;
56 }

68 template <typename LocationT, typename ActionT, typename ConstraintSymbolT>
69 void
70 search_tree_to_graphviz_interactive(
71  const search::SearchTreeNode<LocationT, ActionT, ConstraintSymbolT> *search_node,
72  const std::filesystem::path &output_path,
73  std::istream &is = std::cin)
74 {
75  using Node = search::SearchTreeNode<LocationT, ActionT, ConstraintSymbolT>;
76  std::vector<const Node *> selected_nodes = {search_node};
77  const Node *last_node = search_node;
78  std::function<bool(const Node &)> selector = [&selected_nodes](const Node &node) {
79  return std::find(std::begin(selected_nodes), std::end(selected_nodes), &node)
80  != std::end(selected_nodes);
81  };
82  bool quit = false;
83  Mode mode = Mode::INSERT;
84  fmt::print("Starting interactive debugger\n");
85  print_interactive_help();
86  while (!quit) {
87  fmt::print("Updating output file {} ...", output_path);
88  search_tree_to_graphviz(*search_node, selector).render_to_file(output_path);
89  fmt::print(" done!\n");
90  fmt::print("{}: Please select a child (or 'h' for help):\n", mode);
91 
92  std::map<int, const Node *> selector_map;
93  if (mode == Mode::NAVIGATE) {
94  selector_map =
95  details::create_selector_map<ActionT, Node>(last_node->get_children(), last_node->parents);
96  } else {
97  selector_map = details::create_selector_map<ActionT, Node>(last_node->get_children());
98  }
99 
100  std::string input;
101  std::getline(is, input);
102  if (input == "q") {
103  quit = true;
104  } else if (input == "h") {
105  print_interactive_help();
106  } else if (input == "u") {
107  if (selected_nodes.size() <= 1) {
108  fmt::print("Cannot remove the last node!\n");
109  } else {
110  selected_nodes.pop_back();
111  last_node = selected_nodes.back();
112  }
113  } else if (input == "n") {
114  mode = Mode::NAVIGATE;
115  } else if (input == "i") {
116  mode = Mode::INSERT;
117  } else if (input == "a") {
118  mode = Mode::INSERT_AND_FOLLOW;
119  } else if (mode == Mode::INSERT && input == "*") {
120  for (const auto &[_, child] : selector_map) {
121  selected_nodes.push_back(child);
122  }
123  } else {
124  std::size_t selected = 0;
125  try {
126  selected = std::stoi(input);
127  } catch (std::exception &e) {
128  fmt::print("Failed to parse input '{}': {}\n", input, e.what());
129  continue;
130  }
131  if (selected >= selector_map.size()) {
132  fmt::print("Invalid input (must be in range [0, {}))\n", last_node->get_children().size());
133  continue;
134  }
135  auto &selected_node = selector_map.at(selected);
136  switch (mode) {
137  case Mode::NAVIGATE: last_node = selected_node; break;
138  case Mode::INSERT: selected_nodes.push_back(selected_node); break;
139  case Mode::INSERT_AND_FOLLOW:
140  last_node = selected_node;
141  selected_nodes.push_back(selected_node);
142  break;
143  }
144  }
145  }
146 }
