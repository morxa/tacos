27template <typename LocationT, typename ActionT>
28std::multimap<std::string, automata::ClockConstraint>
29get_constraints_from_time_successor(const search::CanonicalABWord<LocationT, ActionT> &word,
30 RegionIndex max_constant,
31 automata::ta::ConstraintBoundType bound_type)
32{
33 using TARegionState = search::PlantRegionState<LocationT>;
34 std::multimap<std::string, automata::ClockConstraint> res;
35 const auto max_region_index = 2 * max_constant + 1;
36 for (const auto &symbol : word) {
37 for (const auto &region_state : symbol) {
38 assert(std::holds_alternative<TARegionState>(region_state));
39 const TARegionState state = std::get<TARegionState>(region_state);
40 for (const auto &constraint : automata::ta::get_clock_constraints_from_region_index(
41 state.region_index, max_region_index, bound_type)) {
42 res.insert({{state.clock, constraint}});
43 }
44 }
45 }
46 return res;
47}

61template <typename LocationT, typename ActionT, typename ConstraintSymbolT>
62std::multimap<ActionT, std::multimap<std::string, automata::ClockConstraint>>
63get_constraints_from_outgoing_action(
64 const std::set<search::CanonicalABWord<LocationT, ConstraintSymbolT>> canonical_words,
65 const std::pair<RegionIndex, ActionT> & timed_action,
66 RegionIndex K)
67{
68 std::map<ActionT, std::set<RegionIndex>> good_actions;
69 // TODO merging of the constraints is broken because we now get only a single action.
70 good_actions[timed_action.second].insert(timed_action.first);
71
72 // We only need the reg_a of the words. As we know that they are all the same, we can just take
73 // the first one.
74 assert(reg_a(*std::begin(canonical_words)) == reg_a(*std::rbegin(canonical_words)));
75 const auto node_reg_a = reg_a(*std::begin(canonical_words));
76
77 std::multimap<ActionT, std::multimap<std::string, automata::ClockConstraint>> res;
78 for (const auto &[action, increments] : good_actions) {
79 assert(!increments.empty());
80
81 auto first_good_increment = std::begin(increments);
82 for (auto increment = std::begin(increments); increment != std::end(increments); ++increment) {
83 if (std::next(increment) == std::end(increments) || *std::next(increment) > *increment + 1) {
84 std::multimap<std::string, automata::ClockConstraint> constraints;
85 if (first_good_increment == increment) {
86 // They are the same, create both constraints at the same time to obtain a = constraint
87 // for even regions. constraints.merge(
88 constraints.merge(get_constraints_from_time_successor(
89 search::get_nth_time_successor(node_reg_a, *first_good_increment, K),
90 K,
91 automata::ta::ConstraintBoundType::BOTH));
92 } else {
93 constraints.merge(get_constraints_from_time_successor(
94 search::get_nth_time_successor(node_reg_a, *first_good_increment, K),
95 K,
96 automata::ta::ConstraintBoundType::LOWER));
97 constraints.merge(get_constraints_from_time_successor(
98 search::get_nth_time_successor(node_reg_a, *increment, K),
99 K,
100 automata::ta::ConstraintBoundType::UPPER));
101 }
102 res.insert(std::make_pair(action, constraints));
103 first_good_increment = std::next(increment);
104 }
105 }
106 }
107 return res;
108}

110template <typename LocationT, typename ActionT, typename ConstraintSymbolT>
111void
112add_node_to_controller(
113 const search::SearchTreeNode<LocationT, ActionT, ConstraintSymbolT> *const node,
114 std::set<ActionT> controller_actions,
115 std::set<ActionT> environment_actions,
116 RegionIndex K,
117 bool minimize_controller,
118 automata::ta::TimedAutomaton<std::set<search::CanonicalABWord<LocationT, ConstraintSymbolT>>,
119 ActionT> * controller)
120{
121 using search::NodeLabel;
122 using Transition =
123 automata::ta::Transition<std::set<search::CanonicalABWord<LocationT, ConstraintSymbolT>>,
124 ActionT>;
125 using Location =
126 automata::ta::Location<std::set<search::CanonicalABWord<LocationT, ConstraintSymbolT>>>;
127 if (node->label != NodeLabel::TOP) {
128 throw std::invalid_argument(
129 "Cannot create a controller for a node that is not labeled with TOP");
130 }
131 assert(std::is_sorted(std::begin(node->get_children()), std::end(node->get_children())));
132 for (const auto &[timed_action, successor] : node->get_children()) {
133 if (successor->label != NodeLabel::TOP) {
134 continue;
135 }
136 bool new_location = controller->add_location(Location{successor->words});
137 controller->add_final_location(Location{successor->words});
138
139 for (const auto &[action, constraints] :
140 get_constraints_from_outgoing_action(node->words, timed_action, K)) {
141 for (const auto &[clock, _constraint] : constraints) {
142 controller->add_clock(clock);
143 }
144 controller->add_action(action);
145 controller->add_transition(
146 Transition{Location{node->words}, action, Location{successor->words}, constraints, {}});
147 }
148 if (new_location) {
149 // To break circles in the search graph, only add the successor if it is actually a new
150 // location.
151 add_node_to_controller(successor.get(),
152 controller_actions,
153 environment_actions,
154 K,
155 minimize_controller,
156 controller);
157 }
158 if (minimize_controller
159 && controller_actions.find(timed_action.second) != std::end(controller_actions)) {
160 break;
161 }
162 }
163}

167template <typename LocationT, typename ActionT, typename ConstraintSymbolT>
168automata::ta::TimedAutomaton<std::set<search::CanonicalABWord<LocationT, ConstraintSymbolT>>,
169 ActionT>
170create_controller(const search::SearchTreeNode<LocationT, ActionT, ConstraintSymbolT> *const root,
171 std::set<ActionT> controller_actions,
172 std::set<ActionT> environment_actions,
173 RegionIndex K,
174 bool minimize_controller = true)
175{
176 using namespace details;
177 using search::NodeLabel;
178 using Location =
179 automata::ta::Location<std::set<search::CanonicalABWord<LocationT, ConstraintSymbolT>>>;
180 automata::ta::TimedAutomaton<std::set<search::CanonicalABWord<LocationT, ConstraintSymbolT>>,
181 ActionT>
182 controller{{}, Location{root->words}, {}};
183 add_node_to_controller(
184 root, controller_actions, environment_actions, K, minimize_controller, &controller);
185 return controller;
186}
