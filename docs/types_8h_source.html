105template <typename LocationT>
106std::ostream &
107operator<<(std::ostream &os, const PlantConfiguration<LocationT> &configuration)
108{
109 os << "(" << configuration.location << ", ";
110 if (configuration.clock_valuations.empty()) {
111 os << "{})";
112 return os;
113 }
114 os << "{ ";
115 bool first = true;
116 for (const auto &[clock, value] : configuration.clock_valuations) {
117 if (first) {
118 first = false;
119 } else {
120 os << ", ";
121 }
122 os << clock << ": " << value;
123 }
124 os << " } )";
125 return os;
