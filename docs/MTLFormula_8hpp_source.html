44 template <typename APType>
45 std::ostream &
46 operator<<(std::ostream &out, const logic::MTLFormula<APType> &f)
47 {
48  using logic::LOP;
49  using utilities::arithmetic::BoundType;
50  const auto print_until =
51  [](std::ostream &out, const logic::MTLFormula<APType> &f, std::string_view op_symbol) {
52  out << "(" << f.get_operands().front() << " " << op_symbol;
53  const auto &interval = f.get_interval();
54  // Only print interval if it is not (INFTY, INFTY), this corresponds to the usual short
55  // notation used in the literature.
56  if (interval.lowerBoundType() != BoundType::INFTY
57  || interval.upperBoundType() != BoundType::INFTY) {
58  out << interval;
59  }
60  out << " " << f.get_operands().back() << ")";
61  };
62  switch (f.get_operator()) {
63  case LOP::TRUE: out << u8"⊤"; break;
64  case LOP::FALSE: out << u8"⊥"; break;
65  case LOP::AP: out << f.get_atomicProposition(); break;
66  case LOP::LAND:
67  out << "(" << f.get_operands().front() << " ∧ " << f.get_operands().back() << ")";
68  break;
69  case LOP::LOR:
70  out << "(" << f.get_operands().front() << " ∨ " << f.get_operands().back() << ")";
71  break;
72  case LOP::LNEG: out << "!(" << f.get_operands().front() << ")"; break;
73  case LOP::LUNTIL: print_until(out, f, "U"); break;
74  case LOP::LDUNTIL: print_until(out, f, "~U"); break;
75  }
76  return out;
77 }

79 template <typename APType>
80 bool
81 MTLWord<APType>::satisfies_at(const MTLFormula<APType> &phi, std::size_t i) const
82 {
83  if (i >= this->word_.size())
84  return false;
85 
86  switch (phi.operator_) {
87  case LOP::TRUE: return true;
88  case LOP::FALSE: return false;
89  case LOP::AP:
90  return std::find(word_[i].first.begin(), word_[i].first.end(), phi.ap_.value())
91  != word_[i].first.end();
92  break;
93  case LOP::LAND:
94  return std::all_of(phi.operands_.begin(), phi.operands_.end(), [this, i](const auto &subf) {
95  return satisfies_at(subf, i);
96  });
97  break;
98  case LOP::LOR:
99  return std::any_of(phi.operands_.begin(), phi.operands_.end(), [this, i](const auto &subf) {
100  return satisfies_at(subf, i);
101  });
102  break;
103  case LOP::LNEG:
104  return std::none_of(phi.operands_.begin(), phi.operands_.end(), [this, i](const auto &subf) {
105  return satisfies_at(subf, i);
106  });
107  break;
108  case LOP::LUNTIL:
109  for (std::size_t j = i + 1; j < word_.size(); ++j) {
110  // check if termination condition is satisfied, in time.
111  if (satisfies_at(phi.operands_.back(), j)) {
112  assert(phi.duration_.has_value());
113  return phi.duration_.value().contains(word_[j].second - word_[i].second);
114  } else {
115  // check whether first part is satisfied continuously.
116  if (!satisfies_at(phi.operands_.front(), j)) {
117  return false;
118  }
119  }
120  }
121  // if termination condition is never met, return false.
122  return false;
123  break;
124  case LOP::LDUNTIL:
125  assert(phi.duration_.has_value());
126  // using p DU q <=> !(!p U !q) (also called RELEASE operator)
127  // satisfied if:
128  // * q holds always, or
129  // * q holds until (and including this point in time) p becomes true
130  for (std::size_t j = i + 1; j < word_.size(); ++j) {
131  if (satisfies_at(phi.operands_.front(), j) and satisfies_at(phi.operands_.back(), j)) {
132  return phi.duration_.value().contains(word_[j].second - word_[i].second);
133  } else {
134  // check whether q is satisfied (probably indefinitely)
135  if (!satisfies_at(phi.operands_.back(), j)) {
136  return false;
137  }
138  }
139  }
140  // if q holds indefinitely, return true (1st case)
141  return true;
142  break;
143  }
144 
145  return false;
146 }

201 template <typename APType>
202 bool
203 MTLFormula<APType>::operator<(const MTLFormula &rhs) const
204 {
205  // compare operation
206  if (this->get_operator() != rhs.get_operator()) {
207  return this->get_operator() < rhs.get_operator();
208  }
209 
210  // base case: compare atomic propositions
211  if (this->get_operator() == LOP::AP) {
212  assert(rhs.get_operator() == LOP::AP);
213  return this->get_atomicProposition() < rhs.get_atomicProposition();
214  }
215 
216  // compare subformulas
217  // Note: since the operators are the same, the size of operands needs to be the same
218  assert(this->get_operands().size() == rhs.get_operands().size());
219 
220  // Compare intervals before operands.
221  if (operator_ == LOP::LUNTIL || operator_ == LOP::LDUNTIL) {
222  if (duration_ < rhs.duration_) {
223  return true;
224  }
225  if (rhs.duration_ < duration_) {
226  return false;
227  }
228  }
229 
230  return std::lexicographical_compare(this->get_operands().begin(),
231  this->get_operands().end(),
232  rhs.get_operands().begin(),
233  rhs.get_operands().end());
234 }
235 template <typename APType>
236 MTLFormula<APType>
237 MTLFormula<APType>::to_positive_normal_form() const
238 {
239  switch (operator_) {
240  case LOP::TRUE:
241  case LOP::FALSE:
242  case LOP::AP: return *this; break;
243  case LOP::LNEG: {
244  switch (operands_.front().get_operator()) {
245  case LOP::TRUE:
246  case LOP::FALSE:
247  case LOP::AP: return *this; break; // negation in front of ap is conformant
248  case LOP::LNEG:
249  return MTLFormula(operands_.front().get_operands().front())
250  .to_positive_normal_form(); // remove duplicate negations
251  break;
252  case LOP::LAND:
253  case LOP::LOR: {
254  std::vector<MTLFormula<APType>> normalized;
255  for (const auto &op : operands_.front().get_operands()) {
256  normalized.push_back(MTLFormula(LOP::LNEG, {op}).to_positive_normal_form());
257  }
258  return MTLFormula(dual(operands_.front().get_operator()),
259  std::begin(normalized),
260  std::end(normalized));
261  } break;
262  case LOP::LUNTIL:
263  case LOP::LDUNTIL: {
264  // binary operators: negate operands, use dual operator
265  auto neglhs =
266  MTLFormula(LOP::LNEG, {operands_.front().get_operands().front()}).to_positive_normal_form();
267  auto negrhs =
268  MTLFormula(LOP::LNEG, {operands_.front().get_operands().back()}).to_positive_normal_form();
269  return MTLFormula(dual(operands_.front().get_operator()),
270  {neglhs, negrhs},
271  operands_.front().get_interval());
272  } break;
273  }
274  } break;
275  case LOP::LAND:
276  case LOP::LOR:
277  case LOP::LUNTIL:
278  case LOP::LDUNTIL: {
279  std::vector<MTLFormula<APType>> normalized;
280  for (const auto &op : operands_) {
281  normalized.push_back(op.to_positive_normal_form());
282  }
283  auto res = MTLFormula(operator_, std::begin(normalized), std::end(normalized));
284  res.duration_ = duration_;
285  return res;
286  } break;
287  }
288  throw std::logic_error("Error in to_positive_normal_form: should have returned.");
289 }

322 template <typename APType>
323 TimePoint
324 MTLFormula<APType>::get_largest_constant() const
325 {
326  TimePoint largest_constant = 0;
327  switch (operator_) {
328  case LOP::AP:
329  case LOP::TRUE:
330  case LOP::FALSE: largest_constant = 0; break;
331  case LOP::LNEG: largest_constant = operands_[0].get_largest_constant(); break;
332  case LOP::LAND:
333  case LOP::LOR:
334  for (const auto &sub_formula : operands_) {
335  largest_constant = std::max(0., sub_formula.get_largest_constant());
336  }
337  break;
338  case LOP::LUNTIL:
339  case LOP::LDUNTIL: {
340  if (duration_) {
341  if (duration_->upperBoundType() != utilities::arithmetic::BoundType::INFTY) {
342  largest_constant = std::max(largest_constant, duration_->upper());
343  }
344  if (duration_->lowerBoundType() != utilities::arithmetic::BoundType::INFTY) {
345  largest_constant = std::max(largest_constant, duration_->lower());
346  }
347  }
348  largest_constant = std::max(
349  {largest_constant, operands_[0].get_largest_constant(), operands_[1].get_largest_constant()});
350  break;
351  }
352  }
353  return largest_constant;
354 }
