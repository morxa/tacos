16template <typename LocationT>
17TAConfiguration<LocationT>
18get_region_candidate(const RegionalizedConfiguration<LocationT> &regionalized_configuration)
19{
20 TAConfiguration<LocationT> res;
21 res.location = regionalized_configuration.first;
22 std::transform(std::begin(regionalized_configuration.second),
23 std::end(regionalized_configuration.second),
24 std::inserter(res.clock_valuations, res.clock_valuations.end()),
25 [&](const std::pair<std::string, RegionIndex> &clock_region) {
26 const std::string &clock_name = clock_region.first;
27 const RegionIndex &region = clock_region.second;
28 return std::make_pair(clock_name, static_cast<ClockValuation>(region) / 2);
29 });
30 return res;
31}
