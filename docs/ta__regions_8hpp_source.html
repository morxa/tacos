26 template <typename LocationT>
27 Configuration<LocationT>
28 get_region_candidate(const RegionalizedConfiguration<LocationT> &regionalized_configuration)
29 {
30  Configuration<LocationT> res;
31  res.location = regionalized_configuration.first;
32  std::transform(std::begin(regionalized_configuration.second),
33  std::end(regionalized_configuration.second),
34  std::inserter(res.clock_valuations, res.clock_valuations.end()),
35  [&](const std::pair<std::string, RegionIndex> &clock_region) {
36  const std::string &clock_name = clock_region.first;
37  const RegionIndex &region = clock_region.second;
38  return std::make_pair(clock_name, static_cast<ClockValuation>(region) / 2);
39  });
40  return res;
41 }
