40 template <typename LocationT, typename ActionT, typename ConstraintSymbolT>
41 std::optional<utilities::graphviz::Node>
42 add_search_node_to_graph(
43  const search::SearchTreeNode<LocationT, ActionT, ConstraintSymbolT> *search_node,
44  utilities::graphviz::Graph * graph,
45  std::function<bool(const search::SearchTreeNode<LocationT, ActionT, ConstraintSymbolT> &)>
46  node_selector)
47 {
48  if (!node_selector(*search_node)) {
49  return std::nullopt;
50  }
51  std::vector<std::string> words_labels;
52  for (const auto &word : search_node->words) {
53  std::vector<std::string> word_labels;
54  for (const auto &word_partition : word) {
55  std::stringstream str;
56  str << word_partition;
57  word_labels.push_back(str.str());
58  }
59  // Split the partitions into node sections (by using "|" as separator).
60  // Put each word in its own group (with {}) so it is separated from the other words.
61  words_labels.push_back(fmt::format("{{ {} }}", fmt::join(word_labels, "|")));
62  }
63 
64  std::string label_reason;
65  switch (search_node->label_reason) {
66  case LabelReason::UNKNOWN: label_reason = "unknown"; break;
67  case LabelReason::BAD_NODE: label_reason = "bad node"; break;
68  case LabelReason::DEAD_NODE: label_reason = "dead node"; break;
69  case LabelReason::NO_ATA_SUCCESSOR: label_reason = "no ATA successor"; break;
70  case LabelReason::MONOTONIC_DOMINATION: label_reason = "monotonic domination"; break;
71  case LabelReason::NO_BAD_ENV_ACTION: label_reason = "no bad env action"; break;
72  case LabelReason::GOOD_CONTROLLER_ACTION_FIRST:
73  label_reason = "good controller action first";
74  break;
75  case LabelReason::BAD_ENV_ACTION_FIRST: label_reason = "bad env action first"; break;
76  }
77  const std::string node_id = fmt::format("{}", fmt::join(words_labels, "|"));
78  const bool new_node = !graph->has_node(node_id);
79  utilities::graphviz::Node node = graph->get_node(node_id).value_or(
80  graph->add_node(fmt::format("{{{}}}|{}", label_reason, fmt::join(words_labels, "|")), node_id));
81  // Set the node color according to its label.
82  if (search_node->label == search::NodeLabel::TOP) {
83  node.set_property("color", "green");
84  } else if (search_node->label == search::NodeLabel::BOTTOM) {
85  node.set_property("color", "red");
86  }
87  if (new_node) {
88  for (const auto &[action, child] : search_node->get_children()) {
89  auto graphviz_child = add_search_node_to_graph(child.get(), graph, node_selector);
90  if (graphviz_child) {
91  graph->add_edge(node,
92  *graphviz_child,
93  fmt::format("({}, {})", action.first, action.second));
94  }
95  }
96  }
97  return node;
98 }

124 template <typename LocationT, typename ActionT, typename ConstraintSymbolT>
125 utilities::graphviz::Graph
126 search_tree_to_graphviz(
127  const search::SearchTreeNode<LocationT, ActionT, ConstraintSymbolT> &search_node,
128  bool skip_canceled = false)
129 {
130  utilities::graphviz::Graph graph;
131  graph.set_property("rankdir", "LR");
132  graph.set_default_node_property("shape", "record");
133  std::function<bool(const search::SearchTreeNode<LocationT, ActionT, ConstraintSymbolT> &)>
134  selector =
135  [skip_canceled](
136  const search::SearchTreeNode<LocationT, ActionT, ConstraintSymbolT> &node) -> bool {
137  return !skip_canceled || node.label != search::NodeLabel::CANCELED;
138  };
139  add_search_node_to_graph(&search_node, &graph, selector);
140  return graph;
141 }
