31 template <typename LocationT, typename ActionT, typename ConstraintSymbolT>
32 std::optional<utilities::graphviz::Node>
33 add_search_node_to_graph(
34  const search::SearchTreeNode<LocationT, ActionT, ConstraintSymbolT> *search_node,
35  utilities::graphviz::Graph *graph,
36  std::function<bool(const search::SearchTreeNode<LocationT, ActionT, ConstraintSymbolT> &)>
37  node_selector)
38 {
39  if (!node_selector(*search_node)) {
40  return std::nullopt;
41  }
42  std::vector<std::string> words_labels;
43  for (const auto &word : search_node->words) {
44  std::vector<std::string> word_labels;
45  for (const auto &word_partition : word) {
46  std::stringstream str;
47  str << word_partition;
48  word_labels.push_back(str.str());
49  }
50  // Split the partitions into node sections (by using "|" as separator).
51  // Put each word in its own group (with {}) so it is separated from the other words.
52  words_labels.push_back(fmt::format("{{ {} }}", fmt::join(word_labels, "|")));
53  }
54 
55  std::string label_reason;
56  switch (search_node->label_reason) {
57  case LabelReason::UNKNOWN: label_reason = "unknown"; break;
58  case LabelReason::GOOD_NODE: label_reason = "good node"; break;
59  case LabelReason::BAD_NODE: label_reason = "bad node"; break;
60  case LabelReason::DEAD_NODE: label_reason = "dead node"; break;
61  case LabelReason::NO_ATA_SUCCESSOR: label_reason = "no ATA successor"; break;
62  case LabelReason::MONOTONIC_DOMINATION: label_reason = "monotonic domination"; break;
63  case LabelReason::NO_BAD_ENV_ACTION: label_reason = "no bad env action"; break;
64  case LabelReason::GOOD_CONTROLLER_ACTION_FIRST:
65  label_reason = "good controller action first";
66  break;
67  case LabelReason::BAD_ENV_ACTION_FIRST: label_reason = "bad env action first"; break;
68  case LabelReason::ALL_CONTROLLER_ACTIONS_BAD: label_reason = "all ctl actions bad"; break;
69  }
70  const std::string node_id = fmt::format("{}", fmt::join(words_labels, "|"));
71  const bool new_node = !graph->has_node(node_id);
72  utilities::graphviz::Node node = graph->get_node(node_id).value_or(
73  graph->add_node(fmt::format("{{{}}}|{}", label_reason, fmt::join(words_labels, "|")), node_id));
74  // Set the node color according to its label.
75  if (search_node->label == search::NodeLabel::TOP) {
76  node.set_property("color", "green");
77  } else if (search_node->label == search::NodeLabel::BOTTOM) {
78  node.set_property("color", "red");
79  }
80  if (new_node) {
81  for (const auto &[action, child] : search_node->get_children()) {
82  auto graphviz_child = add_search_node_to_graph(child.get(), graph, node_selector);
83  if (graphviz_child) {
84  graph->add_edge(node,
85  *graphviz_child,
86  fmt::format("({}, {})", action.first, action.second));
87  }
88  }
89  }
90  return node;
91 }
