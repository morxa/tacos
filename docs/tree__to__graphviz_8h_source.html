33template <typename LocationT, typename ActionT, typename ConstraintSymbolT>
34std::optional<utilities::graphviz::Node>
35add_search_node_to_graph(
36 const search::SearchTreeNode<LocationT, ActionT, ConstraintSymbolT> *search_node,
37 utilities::graphviz::Graph *graph,
38 std::function<bool(const search::SearchTreeNode<LocationT, ActionT, ConstraintSymbolT> &)>
39 node_selector)
40{
41 if (!node_selector(*search_node)) {
42 return std::nullopt;
43 }
44 std::vector<std::string> words_labels;
45 std::string program_label;
46 for (const auto &word : search_node->words) {
47 std::vector<std::string> word_labels;
48 for (const auto &word_partition : word) {
49 std::vector<std::string> partition_labels;
50 for (const auto &symbol : word_partition) {
51 if (std::holds_alternative<tacos::search::PlantRegionState<LocationT>>(symbol)) {
52 const auto plant_location = std::get<tacos::search::PlantRegionState<LocationT>>(symbol);
53 if (program_label.empty()) {
54 std::stringstream s;
55 s << plant_location.location;
56 program_label = s.str();
57 }
58 partition_labels.push_back(
59 fmt::format("({}, {})", plant_location.clock, plant_location.region_index));
60 } else {
61 std::stringstream str;
62 str << symbol;
63 partition_labels.push_back(str.str());
64 }
65 }
66 word_labels.push_back(fmt::format("{}", fmt::join(partition_labels, ", ")));
67 }
68 // Split the partitions into node sections (by using "|" as separator).
69 // Put each word in its own group (with {}) so it is separated from the other words.
70 words_labels.push_back(fmt::format("{{ {} }}", fmt::join(word_labels, " | ")));
71 }
72
73 std::string label_reason;
74 switch (search_node->label_reason) {
75 case LabelReason::UNKNOWN: label_reason = "unknown"; break;
76 case LabelReason::GOOD_NODE: label_reason = "good node"; break;
77 case LabelReason::BAD_NODE: label_reason = "bad node"; break;
78 case LabelReason::DEAD_NODE: label_reason = "dead node"; break;
79 case LabelReason::NO_ATA_SUCCESSOR: label_reason = "no ATA successor"; break;
80 case LabelReason::MONOTONIC_DOMINATION: label_reason = "monotonic domination"; break;
81 case LabelReason::NO_BAD_ENV_ACTION: label_reason = "no bad env action"; break;
82 case LabelReason::GOOD_CONTROLLER_ACTION_FIRST:
83 label_reason = "good controller action first";
84 break;
85 case LabelReason::BAD_ENV_ACTION_FIRST: label_reason = "bad env action first"; break;
86 case LabelReason::ALL_CONTROLLER_ACTIONS_BAD: label_reason = "all ctl actions bad"; break;
87 }
88 const std::string node_id = fmt::format("{} | {}", program_label, fmt::join(words_labels, " | "));
89 const bool new_node = !graph->has_node(node_id);
90 utilities::graphviz::Node node = graph->get_node(node_id).value_or(graph->add_node(
91 fmt::format("{} | {} | {}", label_reason, program_label, fmt::join(words_labels, " | ")),
92 node_id));
93 // Set the node color according to its label.
94 if (search_node->label == search::NodeLabel::TOP) {
95 node.set_property("color", "green");
96 } else if (search_node->label == search::NodeLabel::BOTTOM) {
97 node.set_property("color", "red");
98 }
99 if (new_node) {
100 for (const auto &[action, child] : search_node->get_children()) {
101 auto graphviz_child = add_search_node_to_graph(child.get(), graph, node_selector);
102 if (graphviz_child) {
103 graph->add_edge(node,
104 *graphviz_child,
105 fmt::format("({}, {})", action.first, action.second));
106 }
107 }
108 }
109 return node;
110}

118template <typename LocationT, typename ActionT, typename ConstraintSymbolT>
119utilities::graphviz::Graph
120search_tree_to_graphviz(
121 const search::SearchTreeNode<LocationT, ActionT, ConstraintSymbolT> &search_node,
122 std::function<bool(const search::SearchTreeNode<LocationT, ActionT, ConstraintSymbolT> &)>
123 node_selector)
124{
125 utilities::graphviz::Graph graph;
126 graph.set_property("rankdir", "LR");
127 graph.set_default_node_property("shape", "record");
128 add_search_node_to_graph(&search_node, &graph, node_selector);
129 return graph;
130}

138template <typename LocationT, typename ActionT, typename ConstraintSymbolT>
139utilities::graphviz::Graph
140search_tree_to_graphviz(
141 const search::SearchTreeNode<LocationT, ActionT, ConstraintSymbolT> &search_node,
142 bool skip_canceled = false)
143{
144 utilities::graphviz::Graph graph;
145 graph.set_property("rankdir", "LR");
146 graph.set_default_node_property("shape", "record");
147 std::function<bool(const search::SearchTreeNode<LocationT, ActionT, ConstraintSymbolT> &)>
148 selector =
149 [skip_canceled](
150 const search::SearchTreeNode<LocationT, ActionT, ConstraintSymbolT> &node) -> bool {
151 return !skip_canceled || node.label != search::NodeLabel::CANCELED;
152 };
153 add_search_node_to_graph(&search_node, &graph, selector);
154 return graph;
155}
