28template <bool use_location_constraints, bool use_set_semantics>
29std::multimap<std::string, CanonicalABWord<GologLocation, std::string>>
30get_next_canonical_words<GologProgram,
31 std::string,
32 std::string,
33 use_location_constraints,
34 use_set_semantics>::
35operator()(
36 const GologProgram &program,
37 const automata::ata::AlternatingTimedAutomaton<logic::MTLFormula<std::string>,
38 logic::AtomicProposition<ATAInputType>> &ata,
39 const std::pair<GologConfiguration, ATAConfiguration<std::string>> &ab_configuration,
40 [[maybe_unused]] const RegionIndex increment,
41 const RegionIndex K)
42{
43 std::multimap<std::string, CanonicalABWord<GologLocation, std::string>> successors;
44 const auto &[_, remaining_program, history] = ab_configuration.first.location;
45 auto golog_successors =
46 program.get_semantics().trans_all(*history,
47 remaining_program.get(),
48 details::get_clock_values(
49 ab_configuration.first.clock_valuations));
50 for (const auto &golog_successor : golog_successors) {
51 // const auto &[plan, program_suffix, new_history] = golog_successor;
52 const auto &plan = std::get<0>(golog_successor);
53 const auto &program_suffix = std::get<1>(golog_successor);
54 const auto &new_history = std::get<2>(golog_successor);
55 const std::string action = plan->elements().front().instruction().str();
56 auto clock_valuations = ab_configuration.first.clock_valuations;
57 if (program.has_action_clock(action)) {
58 std::string prim_action;
59 if (action.substr(0, 6) == "start(") {
60 prim_action = action.substr(6, action.size() - 1 - 6);
61 } else {
62 assert(action.substr(0, 4) == "end(");
63 prim_action = action.substr(4, action.size() - 1 - 4);
64 }
65 // Reset to the clock to 0 if it already exists and otherwise insert a new clock.
66 clock_valuations[prim_action].reset();
67 }
68 if (std::find_if(clock_valuations.begin(),
69 clock_valuations.end(),
70 [](const auto &valuation) { return valuation.first != "golog"; })
71 != std::end(clock_valuations)) {
72 clock_valuations.erase("golog");
73 } else {
74 clock_valuations["golog"] = 0;
75 }
76 const auto ata_successors = [&]() {
77 if constexpr (use_location_constraints) {
78 const auto relevant_satisfied_fluents = [&]() {
79 auto fluents = program.get_relevant_satisfied_fluents(*new_history);
80 // Add the currently occuring action as additional fluent of the form 'occ(action)'.
81 const std::string occ_fluent = fmt::format("occ({})", action);
82 if (program.is_relevant_fluent(occ_fluent)) {
83 fluents.insert(occ_fluent);
84 }
85 return fluents;
86 }();
87 return ata.make_symbol_step(ab_configuration.second, relevant_satisfied_fluents);
88 } else {
89 return ata.make_symbol_step(ab_configuration.second, action);
90 }
91 }();
92 for (const auto &ata_successor : ata_successors) {
93 [[maybe_unused]] auto successor = successors.insert(std::make_pair(
94 action,
95 get_canonical_word(GologConfiguration{{program.get_all_satisfied_fluents(*new_history),
96 program_suffix,
97 new_history},
98 clock_valuations},
99 ata_successor,
100 K)));
101 SPDLOG_TRACE("{}, {}): Getting {} with symbol {}",
102 ab_configuration.first,
103 ab_configuration.second,
104 successor->second,
105 action);
106 }
107 }
108 return successors;
109}
